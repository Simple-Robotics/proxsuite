106#define LDLT_FN_IMPL3(Fn, Prefix, Suffix) \
107 VEG_INLINE static auto Fn(Pack a, Pack b, Pack c) noexcept->Pack \
108 { \
109 return Pack{ simde_mm##Prefix##_##Fn##_##Suffix( \
110 a.inner, b.inner, c.inner) }; \
111 } \
112 VEG_NOM_SEMICOLON

114#define LDLT_ARITHMETIC_IMPL(Prefix, Suffix) \
115 LDLT_FN_IMPL3(fmadd, Prefix, Suffix); /* (a * b + c) */ \
116 LDLT_FN_IMPL3(fnmadd, Prefix, Suffix); /* (-a * b + c) */
117
118#define LDLT_LOAD_STORE(Prefix, Suffix) \
119 VEG_INLINE static auto load_unaligned(ScalarType const* ptr) noexcept->Pack \
120 { \
121 return Pack{ simde_mm##Prefix##_loadu_##Suffix(ptr) }; \
122 } \
123 VEG_INLINE static auto broadcast(ScalarType value) noexcept->Pack \
124 { \
125 return Pack{ simde_mm##Prefix##_set1_##Suffix(value) }; \
126 } \
127 VEG_INLINE void store_unaligned(ScalarType* ptr) const noexcept \
128 { \
129 simde_mm##Prefix##_storeu_##Suffix(ptr, inner); \
130 } \
131 VEG_NOM_SEMICOLON

133template<typename T, usize N>

136template<typename T>
137struct Pack<T, 1>
138{
139 using ScalarType = T;
140
141 T inner;
142
143 VEG_INLINE static auto fmadd(Pack a, Pack b, Pack c) noexcept -> Pack
144 {
145 DENSE_LDLT_FP_PRAGMA
146 return { a.inner * b.inner + c.inner };
147 }
148 VEG_INLINE static auto fnmadd(Pack a, Pack b, Pack c) noexcept -> Pack
149 {
150 return fmadd({ -a.inner }, b, c);
151 }
152 VEG_INLINE static auto load_unaligned(ScalarType const* ptr) noexcept -> Pack
153 {
154 return { *ptr };
155 }
156 VEG_INLINE static auto broadcast(ScalarType value) noexcept -> Pack
157 {
158 return { value };
159 }
160 VEG_INLINE void store_unaligned(ScalarType* ptr) const noexcept
161 {
162 *ptr = inner;
163 }
164};

323template<typename T>
324auto
325adjusted_stride(isize n) noexcept -> isize
326{
327#ifndef SIMDE_NATURAL_FLOAT_VECTOR_SIZE
328 isize simd_stride = 1;
329#else
330 isize simd_stride =
331 (SIMDE_NATURAL_VECTOR_SIZE / CHAR_BIT) / isize{ sizeof(T) };
332#endif
333 return _detail::should_vectorize<T>::value ? _detail::round_up(n, simd_stride)
334 : n;
335}
336template<typename T>
337auto
338align() noexcept -> isize
339{
340 return isize{ alignof(T) } * _detail::adjusted_stride<T>(1);
341}

479 mat.rows(),

481 Eigen::InnerStride<uncvref_t<T>::InnerStrideAtCompileTime>{
482 mat.innerStride(),
483 },
484 };
485 }

488 {
489 return {
490 mat.data() + row_idx * mat.innerStride(),

779template<typename Dst, typename Lhs, typename Rhs, typename T>
780void
781noalias_mul_add_impl(Dst dst, Lhs lhs, Rhs rhs, T factor)
782{
783 VEG_ASSERT_ALL_OF(dst.rows() == lhs.rows(),
784 dst.cols() == rhs.cols(),
785 lhs.cols() == rhs.rows());
786
787 isize nrows = dst.rows();
788 isize ncols = dst.cols();
789 isize depth = lhs.cols();
790
791 if (nrows == 0 || ncols == 0 || depth == 0) {
792 return;
793 }
794
795#if !EIGEN_VERSION_AT_LEAST(3, 3, 8)
796 if ((dst.rows() < 20) && (dst.cols() < 20) && (rhs.rows() < 20)) {
797 // gemm
798 // workaround for eigen 3.3.7 bug:
799 // https://gitlab.com/libeigen/eigen/-/issues/1562
800 using Mat =
801 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::ColMajor, 20, 20>;
802 using MapMut =
803 Eigen::Map<Mat, Eigen::Unaligned, Eigen::OuterStride<Eigen::Dynamic>>;
804
805 auto dst_ =
806 MapMut(dst.data(), dst.rows(), dst.cols(), { dst.outerStride() });
807 dst_.noalias().operator+=(lhs.operator*(rhs).operator*(factor));
808 } else
809#endif
810 {
811 dst.noalias().operator+=(lhs.operator*(rhs).operator*(factor));
812 }
813}
