690solve(
691 optional<SparseMat<T, I>> H,
692 optional<VecRef<T>> g,
693 optional<SparseMat<T, I>> A,
694 optional<VecRef<T>> b,
695 optional<SparseMat<T, I>> C,

709 optional<isize> max_iter = nullopt,
710 proxsuite::proxqp::InitialGuessStatus initial_guess =
711 proxsuite::proxqp::InitialGuessStatus::EQUALITY_CONSTRAINED_INITIAL_GUESS,

713 proxsuite::proxqp::SparseBackend::Automatic)

724 }

734 }

738 if (verbose != nullopt) {
739 Qp.settings.verbose = verbose.value();
740 }
741 if (max_iter != nullopt) {
742 Qp.settings.max_iter = max_iter.value();
743 }

745 Qp.settings.sparse_backend = sparse_backend;
746 Qp.init(H, g, A, b, C, l, u, compute_preconditioner, rho, mu_eq, mu_in);
