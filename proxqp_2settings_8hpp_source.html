18// Sparse backend specifications
19enum struct SparseBackend
20{
21 Automatic, // the solver will select the appropriate sparse backend.
22 SparseCholesky, // sparse cholesky backend.
23 MatrixFree, // iterative matrix free sparse backend.
24};
25
26inline std::ostream&
27operator<<(std::ostream& os, const SparseBackend& sparse_backend)
28{
29 if (sparse_backend == SparseBackend::Automatic)
30 os << "Automatic";
31 else if (sparse_backend == SparseBackend::SparseCholesky) {
32 os << "SparseCholesky";
33 } else {
34 os << "MatrixFree";
35 }
36 return os;
37}
