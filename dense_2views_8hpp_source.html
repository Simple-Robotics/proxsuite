200inline auto
201bytes_to_prev_aligned(void* ptr, usize align) noexcept -> isize
202{
203 using UPtr = std::uintptr_t;
204
205 UPtr mask = align - 1;
206 UPtr iptr = UPtr(ptr);
207 UPtr aligned_ptr = iptr & ~mask;
208 return isize(aligned_ptr - iptr);
209}
210inline auto
211bytes_to_next_aligned(void* ptr, usize align) noexcept -> isize
212{
213 using UPtr = std::uintptr_t;
214
215 UPtr mask = align - 1;
216 UPtr iptr = UPtr(ptr);
217 UPtr aligned_ptr = (iptr + mask) & ~mask;
218 return isize(aligned_ptr - iptr);
219}

1102template<typename T>
1103void
1104noalias_mul_add(MatrixViewMut<T, colmajor> dst,
1105 MatrixView<T, colmajor> lhs,
1106 MatrixView<T, colmajor> rhs,
1107 T factor)
1108{
1109
1110 if ((dst.cols == 0) || (dst.rows == 0) || (lhs.cols == 0)) {
1111 return;
1112 }
1113
1114#if !EIGEN_VERSION_AT_LEAST(3, 3, 8)
1115#define LAZY_PRODUCT(a, b) a.lazyProduct(b)
1116#else
1117#define LAZY_PRODUCT(a, b) a.operator*(b)
1118#endif
1119
1120 if (dst.cols == 1 && dst.rows == 1) {
1121 // dot
1122 auto rhs_col = rhs.col(0);
1123 auto lhs_row = lhs.row(0);
1124 auto lhs_as_col = lhs.col(0);
1125 lhs_as_col.dim = lhs_row.dim;
1126 if (lhs_row.stride == 1) {
1127 dst(0, 0) += factor * lhs_as_col.to_eigen().dot(rhs_col.to_eigen());
1128 } else {
1129 dst(0, 0) += factor * lhs_row.to_eigen().dot(rhs_col.to_eigen());
1130 }
1131 } else if (dst.cols == 1) {
1132 // gemv
1133 auto rhs_col = rhs.col(0);
1134 auto dst_col = dst.col(0);
1135 dst_col.to_eigen().noalias().operator+=(
1136 factor * LAZY_PRODUCT(lhs.to_eigen(), rhs_col.to_eigen()));
1137 }
1138
1139#if !EIGEN_VERSION_AT_LEAST(3, 3, 8)
1140 else if ((dst.rows < 20) && (dst.cols < 20) && (rhs.rows < 20)) {
1141 // gemm
1142 // workaround for eigen 3.3.7 bug:
1143 // https://gitlab.com/libeigen/eigen/-/issues/1562
1144 using Stride = Eigen::OuterStride<Eigen::Dynamic>;
1145 using Mat =
1146 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::ColMajor, 20, 20>;
1147 using MapMut = Eigen::Map<Mat, Eigen::Unaligned, Stride>;
1148 using Map = Eigen::Map<Mat const, Eigen::Unaligned, Stride>;
1149
1150 MapMut(dst.data, dst.rows, dst.cols, Stride(dst.outer_stride))
1151 .noalias()
1152 .
1153 operator+=(
1154 factor *
1155 LAZY_PRODUCT(
1156 Map(lhs.data, lhs.rows, lhs.cols, Stride(lhs.outer_stride)),
1157 Map(rhs.data, rhs.rows, rhs.cols, Stride(rhs.outer_stride))));
1158 }
1159#endif
1160
1161 else {
1162 // gemm
1163 dst.to_eigen().noalias().operator+=(
1164 factor * LAZY_PRODUCT(lhs.to_eigen(), rhs.to_eigen()));
1165 }
1166
1167#undef LAZY_PRODUCT
1168}
