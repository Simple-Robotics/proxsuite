25inline void
26cpuid(std::array<int, 4>& cpui, int level)
27{
28#ifndef _WIN32
29 __cpuid(level, cpui[0], cpui[1], cpui[2], cpui[3]);
30#else
31 __cpuid(cpui.data(), level);
32#endif
33}

35inline void
36cpuidex(std::array<int, 4>& cpui, int level, int count)
37{
38#ifndef _WIN32
39 __cpuid_count(level, count, cpui[0], cpui[1], cpui[2], cpui[3]);
40#else
41 __cpuidex(cpui.data(), level, count);
42#endif
43}

121 Data()
122 : nIds_{ 0 }
123 , nExIds_{ 0 }
124 , isIntel_{ false }
125 , isAMD_{ false }
126 , f_1_ECX_{ 0 }
127 , f_1_EDX_{ 0 }
128 , f_7_EBX_{ 0 }
129 , f_7_ECX_{ 0 }
130 , f_81_ECX_{ 0 }
131 , f_81_EDX_{ 0 }
132 , data_{}
133 , extdata_{}
134 {
135 std::array<int, 4> cpui;
136 typedef unsigned long long bistset_equivalent_type;
137
138 // Calling __cpuid with 0x0 as the function_id argument
139 // gets the number of the highest valid function ID.
140 internal::cpuid(cpui, 0);
141 nIds_ = cpui[0];
142
143 for (int i = 0; i <= nIds_; ++i) {
144 internal::cpuidex(cpui, i, 0);
145 data_.push_back(cpui);
146 }
147
148 // Capture vendor string
149 char vendor[0x20];
150 memset(vendor, 0, sizeof(vendor));
151 *reinterpret_cast<int*>(vendor) = data_[0][1];
152 *reinterpret_cast<int*>(vendor + 4) = data_[0][3];
153 *reinterpret_cast<int*>(vendor + 8) = data_[0][2];
154 vendor_ = vendor;
155 if (vendor_ == "GenuineIntel") {
156 isIntel_ = true;
157 } else if (vendor_ == "AuthenticAMD") {
158 isAMD_ = true;
159 }
160
161 // load bitset with flags for function 0x00000001
162 if (nIds_ >= 1) {
163 f_1_ECX_ = static_cast<bistset_equivalent_type>(data_[1][2]);
164 f_1_EDX_ = static_cast<bistset_equivalent_type>(data_[1][3]);
165 }
166
167 // load bitset with flags for function 0x00000007
168 if (nIds_ >= 7) {
169 f_7_EBX_ = static_cast<bistset_equivalent_type>(data_[7][1]);
170 f_7_ECX_ = static_cast<bistset_equivalent_type>(data_[7][2]);
171 }
172
173 // Calling __cpuid with 0x80000000 as the function_id argument
174 // gets the number of the highest valid extended ID.
175 internal::cpuid(cpui, static_cast<int>(0x80000000));
176 nExIds_ = cpui[0];
177
178 char brand[0x40];
179 memset(brand, 0, sizeof(brand));
180
181 for (int i = static_cast<int>(0x80000000); i <= nExIds_; ++i) {
182 internal::cpuidex(cpui, i, 0);
183 extdata_.push_back(cpui);
184 }
185
186 // load bitset with flags for function 0x80000001
187 if (nExIds_ >= static_cast<int>(0x80000001)) {
188 f_81_ECX_ = static_cast<bistset_equivalent_type>(extdata_[1][2]);
189 f_81_EDX_ = static_cast<bistset_equivalent_type>(extdata_[1][3]);
190 }
191
192 // Interpret CPU brand string if reported
193 if (nExIds_ >= static_cast<int>(0x80000004)) {
194 memcpy(brand, extdata_[2].data(), sizeof(cpui));
195 memcpy(brand + 16, extdata_[3].data(), sizeof(cpui));
196 memcpy(brand + 32, extdata_[4].data(), sizeof(cpui));
197 brand_ = brand;
198 }
199 };
