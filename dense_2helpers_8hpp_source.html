24template<typename T>
25T
26power_iteration(Mat<T>& H,
27 Vec<T>& dw,
28 Vec<T>& rhs,
29 Vec<T>& err_v,
30 T power_iteration_accuracy,
31 isize nb_power_iteration)
32{
33 // computes maximal eigen value of the bottom right matrix of the LDLT
34 isize dim = H.rows();
35 rhs.setZero();
36 // stores eigenvector
37 rhs.array() += 1. / std::sqrt(dim);
38 // stores Hx
39 dw.noalias() = H.template selfadjointView<Eigen::Lower>() * rhs; // Hx
40 T eig = 0;
41 for (isize i = 0; i < nb_power_iteration; i++) {
42
43 rhs = dw / dw.norm();
44 dw.noalias() = (H.template selfadjointView<Eigen::Lower>() * rhs);
45 // calculate associated eigenvalue
46 eig = rhs.dot(dw);
47 // calculate associated error
48 err_v = dw - eig * rhs;
49 T err = proxsuite::proxqp::dense::infty_norm(err_v);
50 // std::cout << "power iteration max: i " << i << " err " << err <<
51 // std::endl;
52 if (err <= power_iteration_accuracy) {
53 break;
54 }
55 }
56 return eig;
57}
58template<typename T>
59T
60min_eigen_value_via_modified_power_iteration(Mat<T>& H,
61 Vec<T>& dw,
62 Vec<T>& rhs,
63 Vec<T>& err_v,
64 T max_eigen_value,
65 T power_iteration_accuracy,
66 isize nb_power_iteration)
67{
68 // performs power iteration on the matrix: max_eigen_value I - H
69 // estimates then the minimal eigenvalue with: minimal_eigenvalue =
70 // max_eigen_value - eig
71 isize dim = H.rows();
72 rhs.setZero();
73 // stores eigenvector
74 rhs.array() += 1. / std::sqrt(dim);
75 // stores Hx
76 dw.noalias() = -(H.template selfadjointView<Eigen::Lower>() * rhs); // Hx
77 dw += max_eigen_value * rhs;
78 T eig = 0;
79 for (isize i = 0; i < nb_power_iteration; i++) {
80
81 rhs = dw / dw.norm();
82 dw.noalias() = -(H.template selfadjointView<Eigen::Lower>() * rhs);
83 dw += max_eigen_value * rhs;
84 // calculate associated eigenvalue
85 eig = rhs.dot(dw);
86 // calculate associated error
87 err_v = dw - eig * rhs;
88 T err = proxsuite::proxqp::dense::infty_norm(err_v);
89 // std::cout << "power iteration min: i " << i << " err " << err <<
90 // std::endl;
91 if (err <= power_iteration_accuracy) {
92 break;
93 }
94 }
95 T minimal_eigenvalue = max_eigen_value - eig;
96 return minimal_eigenvalue;
97}

107template<typename T>
108T
109estimate_minimal_eigen_value_of_symmetric_matrix(
110 Mat<T>& H,
111 EigenValueEstimateMethodOption estimate_method_option,
112 T power_iteration_accuracy,
113 isize nb_power_iteration)
114{
115 PROXSUITE_THROW_PRETTY(
116 (!H.isApprox(H.transpose(), std::numeric_limits<T>::epsilon())),
117 std::invalid_argument,
118 "H is not symmetric.");
119 if (H.size()) {
120 PROXSUITE_CHECK_ARGUMENT_SIZE(
121 H.rows(),
122 H.cols(),
123 "H has a number of rows different of the number of columns.");
124 }
125 isize dim = H.rows();
126 T res(0.);
127 switch (estimate_method_option) {
128 case EigenValueEstimateMethodOption::PowerIteration: {
129 Vec<T> dw(dim);
130 Vec<T> rhs(dim);
131 Vec<T> err(dim);
132 T dominant_eigen_value = power_iteration<T>(
133 H, dw, rhs, err, power_iteration_accuracy, nb_power_iteration);
134 T min_eigenvalue = min_eigen_value_via_modified_power_iteration<T>(
135 H,
136 dw,
137 rhs,
138 err,
139 dominant_eigen_value,
140 power_iteration_accuracy,
141 nb_power_iteration);
142 res = std::min(min_eigenvalue, dominant_eigen_value);
143 } break;
144 case EigenValueEstimateMethodOption::ExactMethod: {
145 Eigen::SelfAdjointEigenSolver<Mat<T>> es(H, Eigen::EigenvaluesOnly);
146 res = T(es.eigenvalues()[0]);
147 } break;
148 }
149 return res;
150}

223template<typename T>
224void
225setup_factorization(Workspace<T>& qpwork,
226 const Model<T>& qpmodel,
227 Results<T>& qpresults,
228 const DenseBackend& dense_backend,
229 const HessianType& hessian_type)
230{
231
232 proxsuite::linalg::veg::dynstack::DynStackMut stack{
233 proxsuite::linalg::veg::from_slice_mut,
234 qpwork.ldl_stack.as_mut(),
235 };
236 switch (hessian_type) {
237 case HessianType::Dense:
238 qpwork.kkt.topLeftCorner(qpmodel.dim, qpmodel.dim) = qpwork.H_scaled;
239 break;
240 case HessianType::Zero:
241 qpwork.kkt.topLeftCorner(qpmodel.dim, qpmodel.dim).setZero();
242 break;
243 case HessianType::Diagonal:
244 qpwork.kkt.topLeftCorner(qpmodel.dim, qpmodel.dim) = qpwork.H_scaled;
245 break;
246 }
247 qpwork.kkt.topLeftCorner(qpmodel.dim, qpmodel.dim).diagonal().array() +=
248 qpresults.info.rho;
249 switch (dense_backend) {
250 case DenseBackend::PrimalDualLDLT:
251 qpwork.kkt.block(0, qpmodel.dim, qpmodel.dim, qpmodel.n_eq) =
252 qpwork.A_scaled.transpose();
253 qpwork.kkt.block(qpmodel.dim, 0, qpmodel.n_eq, qpmodel.dim) =
254 qpwork.A_scaled;
255 qpwork.kkt.bottomRightCorner(qpmodel.n_eq, qpmodel.n_eq).setZero();
256 qpwork.kkt.diagonal()
257 .segment(qpmodel.dim, qpmodel.n_eq)
258 .setConstant(-qpresults.info.mu_eq);
259 qpwork.ldl.factorize(qpwork.kkt.transpose(), stack);
260 break;
261 case DenseBackend::PrimalLDLT:
262 qpwork.kkt.noalias() += qpresults.info.mu_eq_inv *
263 (qpwork.A_scaled.transpose() * qpwork.A_scaled);
264 qpwork.ldl.factorize(qpwork.kkt.transpose(), stack);
265 break;
266 case DenseBackend::Automatic:
267 break;
268 }

356template<typename T>
357void
358update(optional<MatRef<T>> H,
359 optional<VecRef<T>> g,
360 optional<MatRef<T>> A,
361 optional<VecRef<T>> b,
362 optional<MatRef<T>> C,
363 optional<VecRef<T>> l,
364 optional<VecRef<T>> u,
365 optional<VecRef<T>> l_box,
366 optional<VecRef<T>> u_box,
367 Model<T>& model,
368 Workspace<T>& work,
369 const bool box_constraints)
370{
371 // check the model is valid
372 if (g != nullopt) {
373 PROXSUITE_CHECK_ARGUMENT_SIZE(g.value().size(),
374 model.dim,
375 "the dimension wrt the primal variable x "
376 "variable for updating g is not valid.");
377 }
378 if (b != nullopt) {
379 PROXSUITE_CHECK_ARGUMENT_SIZE(b.value().size(),
380 model.n_eq,
381 "the dimension wrt equality constrained "
382 "variables for updating b is not valid.");
383 }
384 if (u != nullopt) {
385 PROXSUITE_CHECK_ARGUMENT_SIZE(u.value().size(),
386 model.n_in,
387 "the dimension wrt inequality constrained "
388 "variables for updating u is not valid.");
389 }
390 if (l != nullopt) {
391 PROXSUITE_CHECK_ARGUMENT_SIZE(l.value().size(),
392 model.n_in,
393 "the dimension wrt inequality constrained "
394 "variables for updating l is not valid.");
395 }
396 if (H != nullopt) {
397 PROXSUITE_CHECK_ARGUMENT_SIZE(
398 H.value().rows(),
399 model.dim,
400 "the row dimension for updating H is not valid.");
401 PROXSUITE_CHECK_ARGUMENT_SIZE(
402 H.value().cols(),
403 model.dim,
404 "the column dimension for updating H is not valid.");
405 }
406 if (A != nullopt) {
407 PROXSUITE_CHECK_ARGUMENT_SIZE(
408 A.value().rows(),
409 model.n_eq,
410 "the row dimension for updating A is not valid.");
411 PROXSUITE_CHECK_ARGUMENT_SIZE(
412 A.value().cols(),
413 model.dim,
414 "the column dimension for updating A is not valid.");
415 }
416 if (C != nullopt) {
417 PROXSUITE_CHECK_ARGUMENT_SIZE(
418 C.value().rows(),
419 model.n_in,
420 "the row dimension for updating C is not valid.");
421 PROXSUITE_CHECK_ARGUMENT_SIZE(
422 C.value().cols(),
423 model.dim,
424 "the column dimension for updating C is not valid.");
425 }
426
427 // update the model
428 if (g != nullopt) {
429 model.g = g.value().eval();
430 }
431 if (b != nullopt) {
432 model.b = b.value().eval();
433 }
434 if (u != nullopt) {
435 model.u = u.value().eval();
436 }
437 if (l != nullopt) {
438 model.l = l.value().eval();
439 }
440 if (u_box != nullopt && box_constraints) {
441 model.u_box = u_box.value();
442 } // else qpmodel.u_box remains initialized to a matrix with zero elements or
443 // zero shape
444
445 if (l_box != nullopt && box_constraints) {
446 model.l_box = l_box.value();
447 } // else qpmodel.l_box remains initialized to a matrix with zero elements or
448 // zero shape
449
450 if (H != nullopt || A != nullopt || C != nullopt) {
451 work.refactorize = true;
452 }
453
454 if (H != nullopt) {
455 model.H = H.value();
456 }
457 if (A != nullopt) {
458 model.A = A.value();
459 }
460 if (C != nullopt) {
461 model.C = C.value();
462 }
463 assert(model.is_valid(box_constraints));
464}
484template<typename T>
485void
486setup( //
487 optional<MatRef<T>> H,
488 optional<VecRef<T>> g,
489 optional<MatRef<T>> A,
490 optional<VecRef<T>> b,
491 optional<MatRef<T>> C,
492 optional<VecRef<T>> l,
493 optional<VecRef<T>> u,
494 optional<VecRef<T>> l_box,
495 optional<VecRef<T>> u_box,
496 Settings<T>& qpsettings,
497 Model<T>& qpmodel,
498 Workspace<T>& qpwork,
499 Results<T>& qpresults,
500 const bool box_constraints,
501 preconditioner::RuizEquilibration<T>& ruiz,
502 PreconditionerStatus preconditioner_status,
503 const HessianType hessian_type)
504{
505
506 switch (qpsettings.initial_guess) {
507 case InitialGuessStatus::EQUALITY_CONSTRAINED_INITIAL_GUESS: {
508 if (qpwork.proximal_parameter_update) {
509 qpresults.cleanup_all_except_prox_parameters();
510 } else {
511 qpresults.cleanup(qpsettings);
512 }
513 qpwork.cleanup(box_constraints);
514 break;
515 }
516 case InitialGuessStatus::COLD_START_WITH_PREVIOUS_RESULT: {
517 // keep solutions but restart workspace and results
518 if (qpwork.proximal_parameter_update) {
519 qpresults.cleanup_statistics();
520 } else {
521 qpresults.cold_start(qpsettings);
522 }
523 qpwork.cleanup(box_constraints);
524 break;
525 }
526 case InitialGuessStatus::NO_INITIAL_GUESS: {
527 if (qpwork.proximal_parameter_update) {
528 qpresults.cleanup_all_except_prox_parameters();
529 } else {
530 qpresults.cleanup(qpsettings);
531 }
532 qpwork.cleanup(box_constraints);
533 break;
534 }
535 case InitialGuessStatus::WARM_START: {
536 if (qpwork.proximal_parameter_update) {
537 qpresults
538 .cleanup_all_except_prox_parameters(); // the warm start is given at
539 // the solve function
540 } else {
541 qpresults.cleanup(qpsettings);
542 }
543 qpwork.cleanup(box_constraints);
544 break;
545 }
546 case InitialGuessStatus::WARM_START_WITH_PREVIOUS_RESULT: {
547 if (qpwork.refactorize || qpwork.proximal_parameter_update) {
548 qpwork.cleanup(box_constraints); // meaningful for when there is an
549 // upate of the model and one wants to
550 // warm start with previous result
551 qpwork.refactorize = true;
552 }
553 qpresults.cleanup_statistics();
554 break;
555 }
556 }
557 if (H != nullopt) {
558 qpmodel.H = H.value();
559 } // else qpmodel.H remains initialzed to a matrix with zero elements
560 if (g != nullopt) {
561 qpmodel.g = g.value();
562 }
563
564 if (A != nullopt) {
565 qpmodel.A = A.value();
566 } // else qpmodel.A remains initialized to a matrix with zero elements or zero
567 // shape
568
569 if (b != nullopt) {
570 qpmodel.b = b.value();
571 } // else qpmodel.b remains initialized to a matrix with zero elements or zero
572 // shape
573
574 if (C != nullopt) {
575 qpmodel.C = C.value();
576 } // else qpmodel.C remains initialized to a matrix with zero elements or zero
577 // shape
578
579 if (u != nullopt) {
580 qpmodel.u = u.value();
581 } // else qpmodel.u remains initialized to a matrix with zero elements or zero
582 // shape
583
584 if (l != nullopt) {
585 qpmodel.l = l.value();
586 } // else qpmodel.l remains initialized to a matrix with zero elements or zero
587 // shape
588 if (u_box != nullopt) {
589 qpmodel.u_box = u_box.value();
590 } // else qpmodel.u_box remains initialized to a matrix with zero elements or
591 // zero shape
592
593 if (l_box != nullopt) {
594 qpmodel.l_box = l_box.value();
595 } // else qpmodel.l_box remains initialized to a matrix with zero elements or
596 // zero shape
597 assert(qpmodel.is_valid(box_constraints));
598 switch (hessian_type) {
599 case HessianType::Zero:
600 break;
601 case HessianType::Dense:
602 qpwork.H_scaled = qpmodel.H;
603 break;
604 case HessianType::Diagonal:
605 qpwork.H_scaled = qpmodel.H;
606 break;
607 }
608 qpwork.g_scaled = qpmodel.g;
609 qpwork.A_scaled = qpmodel.A;
610 qpwork.b_scaled = qpmodel.b;
611 qpwork.C_scaled = qpmodel.C;
612 qpwork.u_scaled =
613 (qpmodel.u.array() <= T(1.E20))
614 .select(qpmodel.u,
615 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.n_in).array() +
616 T(1.E20));
617 qpwork.l_scaled =
618 (qpmodel.l.array() >= T(-1.E20))
619 .select(qpmodel.l,
620 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.n_in).array() -
621 T(1.E20));
622 if (box_constraints) {
623 qpwork.u_box_scaled =
624 (qpmodel.u_box.array() <= T(1.E20))
625 .select(qpmodel.u_box,
626 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.dim).array() +
627 T(1.E20));
628 qpwork.l_box_scaled =
629 (qpmodel.l_box.array() >= T(-1.E20))
630 .select(qpmodel.l_box,
631 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.dim).array() -
632 T(1.E20));
633 }
634
635 qpwork.dual_feasibility_rhs_2 = infty_norm(qpmodel.g);
636 switch (preconditioner_status) {
637 case PreconditionerStatus::EXECUTE:
638 setup_equilibration(
639 qpwork, qpsettings, box_constraints, hessian_type, ruiz, true);
640 break;
641 case PreconditionerStatus::IDENTITY:
642 setup_equilibration(
643 qpwork, qpsettings, box_constraints, hessian_type, ruiz, false);
644 break;
645 case PreconditionerStatus::KEEP:
646 // keep previous one
647 setup_equilibration(
648 qpwork, qpsettings, box_constraints, hessian_type, ruiz, false);
649 break;
650 }

664update_proximal_parameters(Settings<T>& settings,
665 Results<T>& results,
666 Workspace<T>& work,

668 optional<T> mu_eq_new,

702 optional<VecRef<T>> y_wm,
703 optional<VecRef<T>> z_wm,
