73template<typename T>
74void
75setup_factorization(Workspace<T>& qpwork,
76 const Model<T>& qpmodel,
77 Results<T>& qpresults,
78 const DenseBackend& dense_backend,
79 const HessianType& hessian_type)
80{
81
82 proxsuite::linalg::veg::dynstack::DynStackMut stack{
83 proxsuite::linalg::veg::from_slice_mut,
84 qpwork.ldl_stack.as_mut(),
85 };
86 switch (hessian_type) {
87 case HessianType::Dense:
88 qpwork.kkt.topLeftCorner(qpmodel.dim, qpmodel.dim) = qpwork.H_scaled;
89 break;
90 case HessianType::Zero:
91 qpwork.kkt.topLeftCorner(qpmodel.dim, qpmodel.dim).setZero();
92 break;
93 case HessianType::Diagonal:
94 qpwork.kkt.topLeftCorner(qpmodel.dim, qpmodel.dim) = qpwork.H_scaled;
95 break;
96 }
97 qpwork.kkt.topLeftCorner(qpmodel.dim, qpmodel.dim).diagonal().array() +=
98 qpresults.info.rho;
99 switch (dense_backend) {
100 case DenseBackend::PrimalDualLDLT:
101 qpwork.kkt.block(0, qpmodel.dim, qpmodel.dim, qpmodel.n_eq) =
102 qpwork.A_scaled.transpose();
103 qpwork.kkt.block(qpmodel.dim, 0, qpmodel.n_eq, qpmodel.dim) =
104 qpwork.A_scaled;
105 qpwork.kkt.bottomRightCorner(qpmodel.n_eq, qpmodel.n_eq).setZero();
106 qpwork.kkt.diagonal()
107 .segment(qpmodel.dim, qpmodel.n_eq)
108 .setConstant(-qpresults.info.mu_eq);
109 qpwork.ldl.factorize(qpwork.kkt.transpose(), stack);
110 break;
111 case DenseBackend::PrimalLDLT:
112 qpwork.kkt.noalias() += qpresults.info.mu_eq_inv *
113 (qpwork.A_scaled.transpose() * qpwork.A_scaled);
114 qpwork.ldl.factorize(qpwork.kkt.transpose(), stack);
115 break;
116 case DenseBackend::Automatic:
117 break;
118 }
119}
132template<typename T>
133void
134setup_equilibration(Workspace<T>& qpwork,
135 const Settings<T>& qpsettings,
136 const bool box_constraints,
137 const HessianType hessian_type,
138 preconditioner::RuizEquilibration<T>& ruiz,
139 bool execute_preconditioner)
140{
141
142 QpViewBoxMut<T> qp_scaled{
143 { from_eigen, qpwork.H_scaled }, { from_eigen, qpwork.g_scaled },
144 { from_eigen, qpwork.A_scaled }, { from_eigen, qpwork.b_scaled },
145 { from_eigen, qpwork.C_scaled }, { from_eigen, qpwork.u_scaled },
146 { from_eigen, qpwork.l_scaled }, { from_eigen, qpwork.i_scaled },
147 { from_eigen, qpwork.l_box_scaled }, { from_eigen, qpwork.u_box_scaled },
148 };
149
150 proxsuite::linalg::veg::dynstack::DynStackMut stack{
151 proxsuite::linalg::veg::from_slice_mut,
152 qpwork.ldl_stack.as_mut(),
153 };
154 ruiz.scale_qp_in_place(qp_scaled,
155 execute_preconditioner,
156 qpsettings.preconditioner_max_iter,
157 qpsettings.preconditioner_accuracy,
158 hessian_type,
159 box_constraints,
160 qpsettings.primal_infeasibility_solving,
161 stack);
162 qpwork.correction_guess_rhs_g = infty_norm(qpwork.g_scaled);
163}

206template<typename T>
207void
208update(optional<MatRef<T>> H,
209 optional<VecRef<T>> g,
210 optional<MatRef<T>> A,
211 optional<VecRef<T>> b,
212 optional<MatRef<T>> C,
213 optional<VecRef<T>> l,
214 optional<VecRef<T>> u,
215 optional<VecRef<T>> l_box,
216 optional<VecRef<T>> u_box,
217 Model<T>& model,
218 Workspace<T>& work,
219 const bool box_constraints)
220{
221 // check the model is valid
222 if (g != nullopt) {
223 PROXSUITE_CHECK_ARGUMENT_SIZE(g.value().size(),
224 model.dim,
225 "the dimension wrt the primal variable x "
226 "variable for updating g is not valid.");
227 }
228 if (b != nullopt) {
229 PROXSUITE_CHECK_ARGUMENT_SIZE(b.value().size(),
230 model.n_eq,
231 "the dimension wrt equality constrained "
232 "variables for updating b is not valid.");
233 }
234 if (u != nullopt) {
235 PROXSUITE_CHECK_ARGUMENT_SIZE(u.value().size(),
236 model.n_in,
237 "the dimension wrt inequality constrained "
238 "variables for updating u is not valid.");
239 }
240 if (l != nullopt) {
241 PROXSUITE_CHECK_ARGUMENT_SIZE(l.value().size(),
242 model.n_in,
243 "the dimension wrt inequality constrained "
244 "variables for updating l is not valid.");
245 }
246 if (H != nullopt) {
247 PROXSUITE_CHECK_ARGUMENT_SIZE(
248 H.value().rows(),
249 model.dim,
250 "the row dimension for updating H is not valid.");
251 PROXSUITE_CHECK_ARGUMENT_SIZE(
252 H.value().cols(),
253 model.dim,
254 "the column dimension for updating H is not valid.");
255 }
256 if (A != nullopt) {
257 PROXSUITE_CHECK_ARGUMENT_SIZE(
258 A.value().rows(),
259 model.n_eq,
260 "the row dimension for updating A is not valid.");
261 PROXSUITE_CHECK_ARGUMENT_SIZE(
262 A.value().cols(),
263 model.dim,
264 "the column dimension for updating A is not valid.");
265 }
266 if (C != nullopt) {
267 PROXSUITE_CHECK_ARGUMENT_SIZE(
268 C.value().rows(),
269 model.n_in,
270 "the row dimension for updating C is not valid.");
271 PROXSUITE_CHECK_ARGUMENT_SIZE(
272 C.value().cols(),
273 model.dim,
274 "the column dimension for updating C is not valid.");
275 }
276
277 // update the model
278 if (g != nullopt) {
279 model.g = g.value().eval();
280 }
281 if (b != nullopt) {
282 model.b = b.value().eval();
283 }
284 if (u != nullopt) {
285 model.u = u.value().eval();
286 }
287 if (l != nullopt) {
288 model.l = l.value().eval();
289 }
290 if (u_box != nullopt && box_constraints) {
291 model.u_box = u_box.value();
292 } // else qpmodel.u_box remains initialized to a matrix with zero elements or
293 // zero shape
294
295 if (l_box != nullopt && box_constraints) {
296 model.l_box = l_box.value();
297 } // else qpmodel.l_box remains initialized to a matrix with zero elements or
298 // zero shape
299
300 if (H != nullopt || A != nullopt || C != nullopt) {
301 work.refactorize = true;
302 }
303
304 if (H != nullopt) {
305 model.H = H.value();
306 }
307 if (A != nullopt) {
308 model.A = A.value();
309 }
310 if (C != nullopt) {
311 model.C = C.value();
312 }
313 assert(model.is_valid(box_constraints));
314}
334template<typename T>
335void
336setup( //
337 optional<MatRef<T>> H,
338 optional<VecRef<T>> g,
339 optional<MatRef<T>> A,
340 optional<VecRef<T>> b,
341 optional<MatRef<T>> C,
342 optional<VecRef<T>> l,
343 optional<VecRef<T>> u,
344 optional<VecRef<T>> l_box,
345 optional<VecRef<T>> u_box,
346 Settings<T>& qpsettings,
347 Model<T>& qpmodel,
348 Workspace<T>& qpwork,
349 Results<T>& qpresults,
350 const bool box_constraints,
351 preconditioner::RuizEquilibration<T>& ruiz,
352 PreconditionerStatus preconditioner_status,
353 const HessianType hessian_type)
354{
355
356 switch (qpsettings.initial_guess) {
357 case InitialGuessStatus::EQUALITY_CONSTRAINED_INITIAL_GUESS: {
358 if (qpwork.proximal_parameter_update) {
359 qpresults.cleanup_all_except_prox_parameters();
360 } else {
361 qpresults.cleanup(qpsettings);
362 }
363 qpwork.cleanup(box_constraints);
364 break;
365 }
366 case InitialGuessStatus::COLD_START_WITH_PREVIOUS_RESULT: {
367 // keep solutions but restart workspace and results
368 if (qpwork.proximal_parameter_update) {
369 qpresults.cleanup_statistics();
370 } else {
371 qpresults.cold_start(qpsettings);
372 }
373 qpwork.cleanup(box_constraints);
374 break;
375 }
376 case InitialGuessStatus::NO_INITIAL_GUESS: {
377 if (qpwork.proximal_parameter_update) {
378 qpresults.cleanup_all_except_prox_parameters();
379 } else {
380 qpresults.cleanup(qpsettings);
381 }
382 qpwork.cleanup(box_constraints);
383 break;
384 }
385 case InitialGuessStatus::WARM_START: {
386 if (qpwork.proximal_parameter_update) {
387 qpresults
388 .cleanup_all_except_prox_parameters(); // the warm start is given at
389 // the solve function
390 } else {
391 qpresults.cleanup(qpsettings);
392 }
393 qpwork.cleanup(box_constraints);
394 break;
395 }
396 case InitialGuessStatus::WARM_START_WITH_PREVIOUS_RESULT: {
397 if (qpwork.refactorize || qpwork.proximal_parameter_update) {
398 qpwork.cleanup(box_constraints); // meaningful for when there is an
399 // upate of the model and one wants to
400 // warm start with previous result
401 qpwork.refactorize = true;
402 }
403 qpresults.cleanup_statistics();
404 break;
405 }
406 }
407 if (H != nullopt) {
408 qpmodel.H = H.value();
409 } // else qpmodel.H remains initialzed to a matrix with zero elements
410 if (g != nullopt) {
411 qpmodel.g = g.value();
412 }
413
414 if (A != nullopt) {
415 qpmodel.A = A.value();
416 } // else qpmodel.A remains initialized to a matrix with zero elements or zero
417 // shape
418
419 if (b != nullopt) {
420 qpmodel.b = b.value();
421 } // else qpmodel.b remains initialized to a matrix with zero elements or zero
422 // shape
423
424 if (C != nullopt) {
425 qpmodel.C = C.value();
426 } // else qpmodel.C remains initialized to a matrix with zero elements or zero
427 // shape
428
429 if (u != nullopt) {
430 qpmodel.u = u.value();
431 } // else qpmodel.u remains initialized to a matrix with zero elements or zero
432 // shape
433
434 if (l != nullopt) {
435 qpmodel.l = l.value();
436 } // else qpmodel.l remains initialized to a matrix with zero elements or zero
437 // shape
438 if (u_box != nullopt) {
439 qpmodel.u_box = u_box.value();
440 } // else qpmodel.u_box remains initialized to a matrix with zero elements or
441 // zero shape
442
443 if (l_box != nullopt) {
444 qpmodel.l_box = l_box.value();
445 } // else qpmodel.l_box remains initialized to a matrix with zero elements or
446 // zero shape
447 assert(qpmodel.is_valid(box_constraints));
448 switch (hessian_type) {
449 case HessianType::Zero:
450 break;
451 case HessianType::Dense:
452 qpwork.H_scaled = qpmodel.H;
453 break;
454 case HessianType::Diagonal:
455 qpwork.H_scaled = qpmodel.H;
456 break;
457 }
458 qpwork.g_scaled = qpmodel.g;
459 qpwork.A_scaled = qpmodel.A;
460 qpwork.b_scaled = qpmodel.b;
461 qpwork.C_scaled = qpmodel.C;
462 qpwork.u_scaled =
463 (qpmodel.u.array() <= T(1.E20))
464 .select(qpmodel.u,
465 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.n_in).array() +
466 T(1.E20));
467 qpwork.l_scaled =
468 (qpmodel.l.array() >= T(-1.E20))
469 .select(qpmodel.l,
470 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.n_in).array() -
471 T(1.E20));
472 if (box_constraints) {
473 qpwork.u_box_scaled =
474 (qpmodel.u_box.array() <= T(1.E20))
475 .select(qpmodel.u_box,
476 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.dim).array() +
477 T(1.E20));
478 qpwork.l_box_scaled =
479 (qpmodel.l_box.array() >= T(-1.E20))
480 .select(qpmodel.l_box,
481 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.dim).array() -
482 T(1.E20));
483 }
484
485 qpwork.dual_feasibility_rhs_2 = infty_norm(qpmodel.g);
486 switch (preconditioner_status) {
487 case PreconditionerStatus::EXECUTE:
488 setup_equilibration(
489 qpwork, qpsettings, box_constraints, hessian_type, ruiz, true);
490 break;
491 case PreconditionerStatus::IDENTITY:
492 setup_equilibration(
493 qpwork, qpsettings, box_constraints, hessian_type, ruiz, false);
494 break;
495 case PreconditionerStatus::KEEP:
496 // keep previous one
497 setup_equilibration(
498 qpwork, qpsettings, box_constraints, hessian_type, ruiz, false);
499 break;
500 }
501}

512template<typename T>
513void
514update_proximal_parameters(Settings<T>& settings,
515 Results<T>& results,
516 Workspace<T>& work,
517 optional<T> rho_new,
518 optional<T> mu_eq_new,
519 optional<T> mu_in_new)
520{
521
522 if (rho_new != nullopt) {
523 settings.default_rho = rho_new.value();
524 results.info.rho = rho_new.value();
525 work.proximal_parameter_update = true;
526 }
527 if (mu_eq_new != nullopt) {
528 settings.default_mu_eq = mu_eq_new.value();
529 results.info.mu_eq = mu_eq_new.value();
530 results.info.mu_eq_inv = T(1) / results.info.mu_eq;
531 work.proximal_parameter_update = true;
532 }
533 if (mu_in_new != nullopt) {
534 settings.default_mu_in = mu_in_new.value();
535 results.info.mu_in = mu_in_new.value();
536 results.info.mu_in_inv = T(1) / results.info.mu_in;
537 work.proximal_parameter_update = true;
538 }

549template<typename T>
550void
551warm_start(optional<VecRef<T>> x_wm,
552 optional<VecRef<T>> y_wm,
553 optional<VecRef<T>> z_wm,
554 Results<T>& results,
555 Settings<T>& settings,
556 Model<T>& model)
557{
558 if (x_wm == nullopt && y_wm == nullopt && z_wm == nullopt)
559 return;
560
561 settings.initial_guess = InitialGuessStatus::WARM_START;
562
563 // first check problem dimensions
564 if (x_wm != nullopt) {
565 PROXSUITE_CHECK_ARGUMENT_SIZE(
566 x_wm.value().rows(),
567 model.dim,
568 "the dimension wrt primal variable x for warm start is not valid.");
569 }
570
571 if (y_wm != nullopt) {
572 PROXSUITE_CHECK_ARGUMENT_SIZE(y_wm.value().rows(),
573 model.n_eq,
574 "the dimension wrt equality constrained "
575 "variables for warm start is not valid.");
576 }
577
578 if (z_wm != nullopt) {
579 PROXSUITE_CHECK_ARGUMENT_SIZE(
580 z_wm.value().rows(),
581 model.n_in,
582 "the dimension wrt inequality constrained variables for warm start "
583 "is not valid.");
584 }
585
586 if (x_wm != nullopt) {
587 results.x = x_wm.value().eval();
588 }
589
590 if (y_wm != nullopt) {
591 results.y = y_wm.value().eval();
592 }
593
594 if (z_wm != nullopt) {
595 results.z = z_wm.value().eval();
596 }
597}
