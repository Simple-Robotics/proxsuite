67template<typename T>
68void
69setup_factorization(Workspace<T>& qpwork,
70 const Model<T>& qpmodel,
71 Results<T>& qpresults)
72{
73
74 proxsuite::linalg::veg::dynstack::DynStackMut stack{
75 proxsuite::linalg::veg::from_slice_mut,
76 qpwork.ldl_stack.as_mut(),
77 };
78
79 qpwork.kkt.topLeftCorner(qpmodel.dim, qpmodel.dim) = qpwork.H_scaled;
80 qpwork.kkt.topLeftCorner(qpmodel.dim, qpmodel.dim).diagonal().array() +=
81 qpresults.info.rho;
82 qpwork.kkt.block(0, qpmodel.dim, qpmodel.dim, qpmodel.n_eq) =
83 qpwork.A_scaled.transpose();
84 qpwork.kkt.block(qpmodel.dim, 0, qpmodel.n_eq, qpmodel.dim) = qpwork.A_scaled;
85 qpwork.kkt.bottomRightCorner(qpmodel.n_eq, qpmodel.n_eq).setZero();
86 qpwork.kkt.diagonal()
87 .segment(qpmodel.dim, qpmodel.n_eq)
88 .setConstant(-qpresults.info.mu_eq);
89
90 qpwork.ldl.factorize(qpwork.kkt.transpose(), stack);
91}
104template<typename T>
105void
106setup_equilibration(Workspace<T>& qpwork,
107 const Settings<T>& qpsettings,
108 preconditioner::RuizEquilibration<T>& ruiz,
109 bool execute_preconditioner)
110{
111
112 QpViewBoxMut<T> qp_scaled{
113 { from_eigen, qpwork.H_scaled }, { from_eigen, qpwork.g_scaled },
114 { from_eigen, qpwork.A_scaled }, { from_eigen, qpwork.b_scaled },
115 { from_eigen, qpwork.C_scaled }, { from_eigen, qpwork.u_scaled },
116 { from_eigen, qpwork.l_scaled }
117 };
118
119 proxsuite::linalg::veg::dynstack::DynStackMut stack{
120 proxsuite::linalg::veg::from_slice_mut,
121 qpwork.ldl_stack.as_mut(),
122 };
123 ruiz.scale_qp_in_place(qp_scaled,
124 execute_preconditioner,
125 qpsettings.preconditioner_max_iter,
126 qpsettings.preconditioner_accuracy,
127 stack);
128 qpwork.correction_guess_rhs_g = infty_norm(qpwork.g_scaled);
129}

172template<typename T>
173void
174update(optional<MatRef<T>> H,
175 optional<VecRef<T>> g,
176 optional<MatRef<T>> A,
177 optional<VecRef<T>> b,
178 optional<MatRef<T>> C,
179 optional<VecRef<T>> l,
180 optional<VecRef<T>> u,
181 Model<T>& model,
182 Workspace<T>& work)
183{
184 // check the model is valid
185 if (g != nullopt) {
186 PROXSUITE_CHECK_ARGUMENT_SIZE(g.value().rows(),
187 model.dim,
188 "the dimension wrt the primal variable x "
189 "variable for updating g is not valid.");
190 }
191 if (b != nullopt) {
192 PROXSUITE_CHECK_ARGUMENT_SIZE(b.value().rows(),
193 model.n_eq,
194 "the dimension wrt equality constrained "
195 "variables for updating b is not valid.");
196 }
197 if (u != nullopt) {
198 PROXSUITE_CHECK_ARGUMENT_SIZE(u.value().rows(),
199 model.n_in,
200 "the dimension wrt inequality constrained "
201 "variables for updating u is not valid.");
202 }
203 if (l != nullopt) {
204 PROXSUITE_CHECK_ARGUMENT_SIZE(l.value().rows(),
205 model.n_in,
206 "the dimension wrt inequality constrained "
207 "variables for updating l is not valid.");
208 }
209 if (H != nullopt) {
210 PROXSUITE_CHECK_ARGUMENT_SIZE(
211 H.value().rows(),
212 model.dim,
213 "the row dimension for updating H is not valid.");
214 PROXSUITE_CHECK_ARGUMENT_SIZE(
215 H.value().cols(),
216 model.dim,
217 "the column dimension for updating H is not valid.");
218 }
219 if (A != nullopt) {
220 PROXSUITE_CHECK_ARGUMENT_SIZE(
221 A.value().rows(),
222 model.n_eq,
223 "the row dimension for updating A is not valid.");
224 PROXSUITE_CHECK_ARGUMENT_SIZE(
225 A.value().cols(),
226 model.dim,
227 "the column dimension for updating A is not valid.");
228 }
229 if (C != nullopt) {
230 PROXSUITE_CHECK_ARGUMENT_SIZE(
231 C.value().rows(),
232 model.n_in,
233 "the row dimension for updating C is not valid.");
234 PROXSUITE_CHECK_ARGUMENT_SIZE(
235 C.value().cols(),
236 model.dim,
237 "the column dimension for updating C is not valid.");
238 }
239
240 // update the model
241 if (g != nullopt) {
242 model.g = g.value().eval();
243 }
244 if (b != nullopt) {
245 model.b = b.value().eval();
246 }
247 if (u != nullopt) {
248 model.u = u.value().eval();
249 }
250 if (l != nullopt) {
251 model.l = l.value().eval();
252 }
253
254 if (H != nullopt || A != nullopt || C != nullopt) {
255 work.refactorize = true;
256 }
257
258 if (H != nullopt) {
259 model.H = H.value();
260 }
261 if (A != nullopt) {
262 model.A = A.value();
263 }
264 if (C != nullopt) {
265 model.C = C.value();
266 }

287template<typename T>
288void
289setup( //
290 optional<MatRef<T>> H,
291 optional<VecRef<T>> g,
292 optional<MatRef<T>> A,
293 optional<VecRef<T>> b,
294 optional<MatRef<T>> C,
295 optional<VecRef<T>> l,
296 optional<VecRef<T>> u,
297 Settings<T>& qpsettings,
298 Model<T>& qpmodel,
299 Workspace<T>& qpwork,
300 Results<T>& qpresults,
301 preconditioner::RuizEquilibration<T>& ruiz,
302 PreconditionerStatus preconditioner_status)
303{
304
305 switch (qpsettings.initial_guess) {
306 case InitialGuessStatus::EQUALITY_CONSTRAINED_INITIAL_GUESS: {
307 if (qpwork.proximal_parameter_update) {
308 qpresults.cleanup_all_except_prox_parameters();
309 } else {
310 qpresults.cleanup(qpsettings);
311 }
312 qpwork.cleanup();
313 break;
314 }
315 case InitialGuessStatus::COLD_START_WITH_PREVIOUS_RESULT: {
316 // keep solutions but restart workspace and results
317 if (qpwork.proximal_parameter_update) {
318 qpresults.cleanup_statistics();
319 } else {
320 qpresults.cold_start(qpsettings);
321 }
322 qpwork.cleanup();
323 break;
324 }
325 case InitialGuessStatus::NO_INITIAL_GUESS: {
326 if (qpwork.proximal_parameter_update) {
327 qpresults.cleanup_all_except_prox_parameters();
328 } else {
329 qpresults.cleanup(qpsettings);
330 }
331 qpwork.cleanup();
332 break;
333 }
334 case InitialGuessStatus::WARM_START: {
335 if (qpwork.proximal_parameter_update) {
336 qpresults
337 .cleanup_all_except_prox_parameters(); // the warm start is given at
338 // the solve function
339 } else {
340 qpresults.cleanup(qpsettings);
341 }
342 qpwork.cleanup();
343 break;
344 }
345 case InitialGuessStatus::WARM_START_WITH_PREVIOUS_RESULT: {
346 if (qpwork.refactorize || qpwork.proximal_parameter_update) {
347 qpwork.cleanup(); // meaningful for when there is an upate of the model
348 // and one wants to warm start with previous result
349 qpwork.refactorize = true;
350 }
351 qpresults.cleanup_statistics();
352 break;
353 }
354 }
355 if (H != nullopt) {
356 qpmodel.H = H.value();
357 } // else qpmodel.H remains initialzed to a matrix with zero elements
358 if (g != nullopt) {
359 qpmodel.g = g.value();
360 }
361
362 if (A != nullopt) {
363 qpmodel.A = A.value();
364 } // else qpmodel.A remains initialized to a matrix with zero elements or zero
365 // shape
366
367 if (b != nullopt) {
368 qpmodel.b = b.value();
369 } // else qpmodel.b remains initialized to a matrix with zero elements or zero
370 // shape
371
372 if (C != nullopt) {
373 qpmodel.C = C.value();
374 } // else qpmodel.C remains initialized to a matrix with zero elements or zero
375 // shape
376
377 if (u != nullopt) {
378 qpmodel.u = u.value();
379 } // else qpmodel.u remains initialized to a matrix with zero elements or zero
380 // shape
381
382 if (l != nullopt) {
383 qpmodel.l = l.value();
384 } // else qpmodel.l remains initialized to a matrix with zero elements or zero
385 // shape
386
387 qpwork.H_scaled = qpmodel.H;
388 qpwork.g_scaled = qpmodel.g;
389 qpwork.A_scaled = qpmodel.A;
390 qpwork.b_scaled = qpmodel.b;
391 qpwork.C_scaled = qpmodel.C;
392 qpwork.u_scaled =
393 (qpmodel.u.array() <= T(1.E20))
394 .select(qpmodel.u,
395 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.n_in).array() +
396 T(1.E20));
397 qpwork.l_scaled =
398 (qpmodel.l.array() >= T(-1.E20))
399 .select(qpmodel.l,
400 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.n_in).array() -
401 T(1.E20));
402
403 qpwork.dual_feasibility_rhs_2 = infty_norm(qpmodel.g);
404
405 switch (preconditioner_status) {
406 case PreconditionerStatus::EXECUTE:
407 setup_equilibration(qpwork, qpsettings, ruiz, true);
408 break;
409 case PreconditionerStatus::IDENTITY:
410 setup_equilibration(qpwork, qpsettings, ruiz, false);
411 break;
412 case PreconditionerStatus::KEEP:
413 // keep previous one
414 setup_equilibration(qpwork, qpsettings, ruiz, false);
415 break;
416 }
417}

428template<typename T>
429void
430update_proximal_parameters(Settings<T>& settings,
431 Results<T>& results,
432 Workspace<T>& work,
433 optional<T> rho_new,
434 optional<T> mu_eq_new,
435 optional<T> mu_in_new)
436{
437
438 if (rho_new != nullopt) {
439 settings.default_rho = rho_new.value();
440 results.info.rho = rho_new.value();
441 work.proximal_parameter_update = true;
442 }
443 if (mu_eq_new != nullopt) {
444 settings.default_mu_eq = mu_eq_new.value();
445 results.info.mu_eq = mu_eq_new.value();
446 results.info.mu_eq_inv = T(1) / results.info.mu_eq;
447 work.proximal_parameter_update = true;
448 }
449 if (mu_in_new != nullopt) {
450 settings.default_mu_in = mu_in_new.value();
451 results.info.mu_in = mu_in_new.value();
452 results.info.mu_in_inv = T(1) / results.info.mu_in;
453 work.proximal_parameter_update = true;
454 }

465template<typename T>
466void
467warm_start(optional<VecRef<T>> x_wm,
468 optional<VecRef<T>> y_wm,
469 optional<VecRef<T>> z_wm,
470 Results<T>& results,
471 Settings<T>& settings,
472 Model<T>& model)
473{
474 if (x_wm == nullopt && y_wm == nullopt && z_wm == nullopt)
475 return;
476
477 settings.initial_guess = InitialGuessStatus::WARM_START;
478
479 // first check problem dimensions
480 if (x_wm != nullopt) {
481 PROXSUITE_CHECK_ARGUMENT_SIZE(
482 x_wm.value().rows(),
483 model.dim,
484 "the dimension wrt primal variable x for warm start is not valid.");
485 }
486
487 if (y_wm != nullopt) {
488 PROXSUITE_CHECK_ARGUMENT_SIZE(y_wm.value().rows(),
489 model.n_eq,
490 "the dimension wrt equality constrained "
491 "variables for warm start is not valid.");
492 }
493
494 if (z_wm != nullopt) {
495 PROXSUITE_CHECK_ARGUMENT_SIZE(
496 z_wm.value().rows(),
497 model.n_in,
498 "the dimension wrt inequality constrained variables for warm start "
499 "is not valid.");
500 }
501
502 if (x_wm != nullopt) {
503 results.x = x_wm.value().eval();
504 }
505
506 if (y_wm != nullopt) {
507 results.y = y_wm.value().eval();
508 }
509
510 if (z_wm != nullopt) {
511 results.z = z_wm.value().eval();
512 }
513}
