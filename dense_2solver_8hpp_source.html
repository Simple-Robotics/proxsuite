38template<typename T>
39void
40refactorize(const Model<T>& qpmodel,
41 Results<T>& qpresults,
42 Workspace<T>& qpwork,
43 T rho_new)
44{
45
46 if (!qpwork.constraints_changed && rho_new == qpresults.info.rho) {
47 return;
48 }
49
50 qpwork.dw_aug.setZero();
51 qpwork.kkt.diagonal().head(qpmodel.dim).array() +=
52 rho_new - qpresults.info.rho;
53 qpwork.kkt.diagonal().segment(qpmodel.dim, qpmodel.n_eq).array() =
54 -qpresults.info.mu_eq;
55
56 proxsuite::linalg::veg::dynstack::DynStackMut stack{
57 proxsuite::linalg::veg::from_slice_mut, qpwork.ldl_stack.as_mut()
58 };
59 qpwork.ldl.factorize(qpwork.kkt.transpose(), stack);
60
61 isize n = qpmodel.dim;
62 isize n_eq = qpmodel.n_eq;
63 isize n_in = qpmodel.n_in;
64 isize n_c = qpwork.n_c;
65
66 LDLT_TEMP_MAT(T, new_cols, n + n_eq + n_c, n_c, stack);
67 T mu_in_neg = -qpresults.info.mu_in;
68 for (isize i = 0; i < n_in; ++i) {
69 isize j = qpwork.current_bijection_map[i];
70 if (j < n_c) {
71 auto col = new_cols.col(j);
72 col.head(n) = qpwork.C_scaled.row(i);
73 col.segment(n, n_eq + n_c).setZero();
74 col(n + n_eq + j) = mu_in_neg;
75 }
76 }
77 qpwork.ldl.insert_block_at(n + n_eq, new_cols, stack);
78
79 qpwork.constraints_changed = false;
80
81 qpwork.dw_aug.setZero();
82}

95template<typename T>
96void
97mu_update(const Model<T>& qpmodel,
98 Results<T>& qpresults,
99 Workspace<T>& qpwork,
100 T mu_eq_new,
101 T mu_in_new)
102{
103 proxsuite::linalg::veg::dynstack::DynStackMut stack{
104 proxsuite::linalg::veg::from_slice_mut, qpwork.ldl_stack.as_mut()
105 };
106
107 isize n = qpmodel.dim;
108 isize n_eq = qpmodel.n_eq;
109 isize n_c = qpwork.n_c;
110
111 if ((n_eq + n_c) == 0) {
112 return;
113 }
114
115 LDLT_TEMP_VEC_UNINIT(T, rank_update_alpha, n_eq + n_c, stack);
116 rank_update_alpha.head(n_eq).setConstant(qpresults.info.mu_eq - mu_eq_new);
117 rank_update_alpha.tail(n_c).setConstant(qpresults.info.mu_in - mu_in_new);
118
119 {
120 auto _indices = stack.make_new_for_overwrite(
121 proxsuite::linalg::veg::Tag<isize>{}, n_eq + n_c);
122 isize* indices = _indices.ptr_mut();
123 for (isize k = 0; k < n_eq; ++k) {
124 indices[k] = n + k;
125 }
126 for (isize k = 0; k < n_c; ++k) {
127 indices[n_eq + k] = n + n_eq + k;
128 }
129 qpwork.ldl.diagonal_update_clobber_indices(
130 indices, n_eq + n_c, rank_update_alpha, stack);
131 }
132
133 qpwork.constraints_changed = true;
134}
145template<typename T>
146void
147iterative_residual(const Model<T>& qpmodel,
148 Results<T>& qpresults,
149 Workspace<T>& qpwork,
150 isize inner_pb_dim)
151{
152
153 qpwork.err.head(inner_pb_dim) = qpwork.rhs.head(inner_pb_dim);
154
155 qpwork.err.head(qpmodel.dim).noalias() -=
156 qpwork.H_scaled.template selfadjointView<Eigen::Lower>() *
157 qpwork.dw_aug.head(qpmodel.dim);
158 qpwork.err.head(qpmodel.dim) -=
159 qpresults.info.rho * qpwork.dw_aug.head(qpmodel.dim);
160
161 // PERF: fuse {A, C}_scaled multiplication operations
162 qpwork.err.head(qpmodel.dim).noalias() -=
163 qpwork.A_scaled.transpose() *
164 qpwork.dw_aug.segment(qpmodel.dim, qpmodel.n_eq);
165 for (isize i = 0; i < qpmodel.n_in; i++) {
166 isize j = qpwork.current_bijection_map(i);
167 if (j < qpwork.n_c) {
168 qpwork.err.head(qpmodel.dim).noalias() -=
169 qpwork.dw_aug(qpmodel.dim + qpmodel.n_eq + j) * qpwork.C_scaled.row(i);
170 qpwork.err(qpmodel.dim + qpmodel.n_eq + j) -=
171 (qpwork.C_scaled.row(i).dot(qpwork.dw_aug.head(qpmodel.dim)) -
172 qpwork.dw_aug(qpmodel.dim + qpmodel.n_eq + j) * qpresults.info.mu_in);
173 }
174 }
175 qpwork.err.segment(qpmodel.dim, qpmodel.n_eq).noalias() -=
176 qpwork.A_scaled * qpwork.dw_aug.head(qpmodel.dim);
177 qpwork.err.segment(qpmodel.dim, qpmodel.n_eq) +=
178 qpwork.dw_aug.segment(qpmodel.dim, qpmodel.n_eq) * qpresults.info.mu_eq;
179}
192template<typename T>
193void
194iterative_solve_with_permut_fact( //
195 const Settings<T>& qpsettings,
196 const Model<T>& qpmodel,
197 Results<T>& qpresults,
198 Workspace<T>& qpwork,
199 T eps,
200 isize inner_pb_dim)
201{
202
203 qpwork.err.setZero();
204 i32 it = 0;
205 i32 it_stability = 0;
206
207 qpwork.dw_aug.head(inner_pb_dim) = qpwork.rhs.head(inner_pb_dim);
208 proxsuite::linalg::veg::dynstack::DynStackMut stack{
209 proxsuite::linalg::veg::from_slice_mut, qpwork.ldl_stack.as_mut()
210 };
211 qpwork.ldl.solve_in_place(qpwork.dw_aug.head(inner_pb_dim), stack);
212
213 iterative_residual<T>(qpmodel, qpresults, qpwork, inner_pb_dim);
214
215 ++it;
216 T preverr = infty_norm(qpwork.err.head(inner_pb_dim));
217 /* to put in debuger mode
218 if (qpsettings.verbose) {
219 std::cout << "infty_norm(res) " <<
220 infty_norm(qpwork.err.head(inner_pb_dim))
221 << std::endl;
222 }
223 */
224 while (infty_norm(qpwork.err.head(inner_pb_dim)) >= eps) {
225
226 if (it >= qpsettings.nb_iterative_refinement) {
227 break;
228 }
229
230 ++it;
231 qpwork.ldl.solve_in_place(qpwork.err.head(inner_pb_dim), stack);
232 qpwork.dw_aug.head(inner_pb_dim) += qpwork.err.head(inner_pb_dim);
233
234 qpwork.err.head(inner_pb_dim).setZero();
235 iterative_residual<T>(qpmodel, qpresults, qpwork, inner_pb_dim);
236
237 if (infty_norm(qpwork.err.head(inner_pb_dim)) > preverr) {
238 it_stability += 1;
239
240 } else {
241 it_stability = 0;
242 }
243 if (it_stability == 2) {
244 break;
245 }
246 preverr = infty_norm(qpwork.err.head(inner_pb_dim));
247 /* to put in debug mode
248 if (qpsettings.verbose) {
249 std::cout << "infty_norm(res) "
250 <<
251 infty_norm(qpwork.err.head(inner_pb_dim)) << std::endl;
252 }
253 */
254 }
255
256 if (infty_norm(qpwork.err.head(inner_pb_dim)) >=
257 std::max(eps, qpsettings.eps_refact)) {
258 refactorize(qpmodel, qpresults, qpwork, qpresults.info.rho);
259 it = 0;
260 it_stability = 0;
261
262 qpwork.dw_aug.head(inner_pb_dim) = qpwork.rhs.head(inner_pb_dim);
263 qpwork.ldl.solve_in_place(qpwork.dw_aug.head(inner_pb_dim), stack);
264
265 iterative_residual<T>(qpmodel, qpresults, qpwork, inner_pb_dim);
266
267 preverr = infty_norm(qpwork.err.head(inner_pb_dim));
268 ++it;
269 /* to put in debug mode
270 if (qpsettings.verbose) {
271 std::cout << "infty_norm(res) "
272 <<
273 infty_norm(qpwork.err.head(inner_pb_dim)) << std::endl;
274 }
275 */
276 while (infty_norm(qpwork.err.head(inner_pb_dim)) >= eps) {
277
278 if (it >= qpsettings.nb_iterative_refinement) {
279 break;
280 }
281 ++it;
282 qpwork.ldl.solve_in_place(qpwork.err.head(inner_pb_dim), stack);
283 qpwork.dw_aug.head(inner_pb_dim) += qpwork.err.head(inner_pb_dim);
284
285 qpwork.err.head(inner_pb_dim).setZero();
286 iterative_residual<T>(qpmodel, qpresults, qpwork, inner_pb_dim);
287
288 if (infty_norm(qpwork.err.head(inner_pb_dim)) > preverr) {
289 it_stability += 1;
290
291 } else {
292 it_stability = 0;
293 }
294 if (it_stability == 2) {
295 break;
296 }
297 preverr = infty_norm(qpwork.err.head(inner_pb_dim));
298 /* to put in debug mode
299 if (qpsettings.verbose) {
300 std::cout << "infty_norm(res) "
301 <<
302 infty_norm(qpwork.err.head(inner_pb_dim)) << std::endl;
303 }
304 */
305 }
306 }
307 qpwork.rhs.head(inner_pb_dim).setZero();
308}

454template<typename T>
455auto
456compute_inner_loop_saddle_point(const Model<T>& qpmodel,
457 Results<T>& qpresults,
458 Workspace<T>& qpwork) -> T
459{
460
461 qpwork.active_part_z =
462 helpers::positive_part(qpwork.primal_residual_in_scaled_up) +
463 helpers::negative_part(qpwork.primal_residual_in_scaled_low) -
464 qpresults.z * qpresults.info.mu_in; // contains now : [Cx-u+z_prev*mu_in]+
465 // + [Cx-l+z_prev*mu_in]- - z*mu_in
466
467 T err = infty_norm(qpwork.active_part_z);
468 qpwork.err.segment(qpmodel.dim, qpmodel.n_eq) =
469 qpwork.primal_residual_eq_scaled; // contains now Ax-b-(y-y_prev)/mu
470
471 T prim_eq_e = infty_norm(
472 qpwork.err.segment(qpmodel.dim, qpmodel.n_eq)); // ||Ax-b-(y-y_prev)/mu||
473 err = std::max(err, prim_eq_e);
474 T dual_e =
475 infty_norm(qpwork.dual_residual_scaled); // contains ||Hx + rho(x-xprev) +
476 // g + Aty + Ctz||
477 err = std::max(err, dual_e);
478
479 return err;
480}
491template<typename T>
492void
493primal_dual_semi_smooth_newton_step(const Settings<T>& qpsettings,
494 const Model<T>& qpmodel,
495 Results<T>& qpresults,
496 Workspace<T>& qpwork,
497 T eps)
498{
499
500 /* MUST BE
501 * dual_residual_scaled = Hx + rho * (x-x_prev) + A.T y + C.T z
502 * primal_residual_eq_scaled = Ax-b+mu_eq (y_prev-y)
503 * primal_residual_in_scaled_up = Cx-u+mu_in(z_prev)
504 * primal_residual_in_scaled_low = Cx-l+mu_in(z_prev)
505 */
506
507 qpwork.active_set_up.array() =
508 (qpwork.primal_residual_in_scaled_up.array() >= 0);
509 qpwork.active_set_low.array() =
510 (qpwork.primal_residual_in_scaled_low.array() <= 0);
511 qpwork.active_inequalities = qpwork.active_set_up || qpwork.active_set_low;
512 isize numactive_inequalities = qpwork.active_inequalities.count();
513
514 isize inner_pb_dim = qpmodel.dim + qpmodel.n_eq + numactive_inequalities;
515 qpwork.rhs.setZero();
516 qpwork.dw_aug.setZero();
517
518 linesearch::active_set_change(qpmodel, qpresults, qpwork);
519
520 qpwork.rhs.head(qpmodel.dim) = -qpwork.dual_residual_scaled;
521
522 qpwork.rhs.segment(qpmodel.dim, qpmodel.n_eq) =
523 -qpwork.primal_residual_eq_scaled;
524 for (isize i = 0; i < qpmodel.n_in; i++) {
525 isize j = qpwork.current_bijection_map(i);
526 if (j < qpwork.n_c) {
527 if (qpwork.active_set_up(i)) {
528 qpwork.rhs(j + qpmodel.dim + qpmodel.n_eq) =
529 -qpwork.primal_residual_in_scaled_up(i) +
530 qpresults.z(i) * qpresults.info.mu_in;
531 } else if (qpwork.active_set_low(i)) {
532 qpwork.rhs(j + qpmodel.dim + qpmodel.n_eq) =
533 -qpwork.primal_residual_in_scaled_low(i) +
534 qpresults.z(i) * qpresults.info.mu_in;
535 }
536 } else {
537 qpwork.rhs.head(qpmodel.dim) +=
538 qpresults.z(i) * qpwork.C_scaled.row(i); // unactive unrelevant columns
539 }
540 }
541
542 iterative_solve_with_permut_fact( //
543 qpsettings,
544 qpmodel,
545 qpresults,
546 qpwork,
547 eps,
548 inner_pb_dim);
549
550 // use active_part_z as a temporary variable to derive unpermutted dz step
551 for (isize j = 0; j < qpmodel.n_in; ++j) {
552 isize i = qpwork.current_bijection_map(j);
553 if (i < qpwork.n_c) {
554 qpwork.active_part_z(j) = qpwork.dw_aug(qpmodel.dim + qpmodel.n_eq + i);
555 } else {
556 qpwork.active_part_z(j) = -qpresults.z(j);
557 }
558 }
559 qpwork.dw_aug.tail(qpmodel.n_in) = qpwork.active_part_z;
560}
573template<typename T>
574void
575primal_dual_newton_semi_smooth(const Settings<T>& qpsettings,
576 const Model<T>& qpmodel,
577 Results<T>& qpresults,
578 Workspace<T>& qpwork,
579 preconditioner::RuizEquilibration<T>& ruiz,
580 T eps_int)
581{
582
583 /* MUST CONTAIN IN ENTRY WITH x = x_prev ; y = y_prev ; z = z_prev
584 * dual_residual_scaled = Hx + rho * (x-x_prev) + A.T y + C.T z
585 * primal_residual_eq_scaled = Ax-b+mu_eq (y_prev-y)
586 * primal_residual_in_scaled_up = Cx-u+mu_in(z_prev)
587 * primal_residual_in_scaled_low = Cx-l+mu_in(z_prev)
588 */
589 /* for debug
590 if (qpsettings.verbose) {
591 std::cout << "---- inner iteration inner error alpha ----" <<
592 std::endl;
593 }
594 */
595 T err_in = 1.e6;
596
597 for (i64 iter = 0; iter <= qpsettings.max_iter_in; ++iter) {
598
599 if (iter == qpsettings.max_iter_in) {
600 qpresults.info.iter += qpsettings.max_iter_in + 1;
601 break;
602 }
603 primal_dual_semi_smooth_newton_step<T>(
604 qpsettings, qpmodel, qpresults, qpwork, eps_int);
605
606 proxsuite::linalg::veg::dynstack::DynStackMut stack{
607 proxsuite::linalg::veg::from_slice_mut, qpwork.ldl_stack.as_mut()
608 };
609 LDLT_TEMP_VEC(T, ATdy, qpmodel.dim, stack);
610 LDLT_TEMP_VEC(T, CTdz, qpmodel.dim, stack);
611
612 auto& Hdx = qpwork.Hdx;
613 auto& Adx = qpwork.Adx;
614 auto& Cdx = qpwork.Cdx;
615
616 auto dx = qpwork.dw_aug.head(qpmodel.dim);
617 auto dy = qpwork.dw_aug.segment(qpmodel.dim, qpmodel.n_eq);
618 auto dz = qpwork.dw_aug.segment(qpmodel.dim + qpmodel.n_eq, qpmodel.n_in);
619
620 Hdx.setZero();
621 Adx.setZero();
622 Cdx.setZero();
623
624 Hdx.noalias() +=
625 qpwork.H_scaled.template selfadjointView<Eigen::Lower>() * dx;
626
627 Adx.noalias() += qpwork.A_scaled * dx;
628 ATdy.noalias() += qpwork.A_scaled.transpose() * dy;
629
630 Cdx.noalias() += qpwork.C_scaled * dx;
631 CTdz.noalias() += qpwork.C_scaled.transpose() * dz;
632
633 if (qpmodel.n_in > 0) {
634 linesearch::primal_dual_ls(qpmodel, qpresults, qpwork);
635 }
636 auto alpha = qpwork.alpha;
637
638 if (infty_norm(alpha * qpwork.dw_aug) < 1.E-11 && iter > 0) {
639 qpresults.info.iter += iter + 1;
640 /* to put in debuger mode
641 if (qpsettings.verbose) {
642 std::cout << "infty_norm(alpha_step * dx) "
643 << infty_norm(alpha *
644 qpwork.dw_aug) << std::endl;
645 }
646 */
647 break;
648 }
649
650 qpresults.x += alpha * dx;
651
652 // contains now : C(x+alpha dx)-u + z_prev * mu_in
653 qpwork.primal_residual_in_scaled_up += alpha * Cdx;
654
655 // contains now : C(x+alpha dx)-l + z_prev * mu_in
656 qpwork.primal_residual_in_scaled_low += alpha * Cdx;
657
658 qpwork.primal_residual_eq_scaled +=
659 alpha * (Adx - qpresults.info.mu_eq * dy);
660
661 qpresults.y += alpha * dy;
662 qpresults.z += alpha * dz;
663
664 qpwork.dual_residual_scaled +=
665 alpha * (qpresults.info.rho * dx + Hdx + ATdy + CTdz);
666
667 err_in = dense::compute_inner_loop_saddle_point(qpmodel, qpresults, qpwork);
668 /* for debug
669 if (qpsettings.verbose) {
670 std::cout << " " << iter << " " <<
671 std::setprecision(2) << err_in
672 << " " << alpha <<
673 std::endl;
674 }
675 */
676 if (qpsettings.verbose) {
677 std::cout << "\033[1;34m[inner iteration " << iter + 1 << "]\033[0m"
678 << std::endl;
679 std::cout << std::scientific << std::setw(2) << std::setprecision(2)
680 << "| inner residual=" << err_in << " | alpha=" << alpha
681 << std::endl;
682 }
683 if (err_in <= eps_int) {
684 qpresults.info.iter += iter + 1;
685 /* for debug
686 if (qpsettings.verbose) {
687 std::cout << "-------------------------------------------------"
688 << std::endl;
689 }
690 */
691 break;
692 }
693
694 // compute primal and dual infeasibility criteria
695 bool is_primal_infeasible =
696 global_primal_residual_infeasibility(VectorViewMut<T>{ from_eigen, ATdy },
697 VectorViewMut<T>{ from_eigen, CTdz },
698 VectorViewMut<T>{ from_eigen, dy },
699 VectorViewMut<T>{ from_eigen, dz },
700 qpwork,
701 qpsettings,
702 ruiz);
703
704 bool is_dual_infeasible =
