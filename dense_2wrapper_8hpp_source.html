515
516 if (eps_abs != std::nullopt) {
517 Qp.settings.eps_abs = eps_abs.value();
518 }
519 if (eps_rel != std::nullopt) {
520 Qp.settings.eps_rel = eps_rel.value();
521 }
522 if (verbose != std::nullopt) {
523 Qp.settings.verbose = verbose.value();
524 }
525 if (max_iter != std::nullopt) {
526 Qp.settings.max_iter = verbose.value();
527 }
528 Qp.settings.compute_timings = compute_timings;
