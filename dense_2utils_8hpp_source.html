134template<typename T>
135void
136global_primal_residual(const Model<T>& qpmodel,
137 const Results<T>& qpresults,
138 Workspace<T>& qpwork,
139 const preconditioner::RuizEquilibration<T>& ruiz,
140 T& primal_feasibility_lhs,
141 T& primal_feasibility_eq_rhs_0,
142 T& primal_feasibility_in_rhs_0,
143 T& primal_feasibility_eq_lhs,
144 T& primal_feasibility_in_lhs)
145{
146 // COMPUTES:
147 // primal_residual_eq_scaled = scaled(Ax - b)
148 //
149 // primal_feasibility_lhs = max(norm(unscaled(Ax - b)),
150 // norm(unscaled([Cx - u]+ + [Cx - l]-)))
151 // primal_feasibility_eq_rhs_0 = norm(unscaled(Ax))
152 // primal_feasibility_in_rhs_0 = norm(unscaled(Cx))
153 //
154 // MAY_ALIAS[primal_residual_in_scaled_u, primal_residual_in_scaled_l]
155 //
156 // INDETERMINATE:
157 // primal_residual_in_scaled_u = unscaled(Cx)
158 // primal_residual_in_scaled_l = unscaled([Cx - u]+ + [Cx - l]-)
159 qpwork.primal_residual_eq_scaled.noalias() = qpwork.A_scaled * qpresults.x;
160 qpwork.primal_residual_in_scaled_up.noalias() = qpwork.C_scaled * qpresults.x;
161
162 ruiz.unscale_primal_residual_in_place_eq(
163 VectorViewMut<T>{ from_eigen, qpwork.primal_residual_eq_scaled });
164 primal_feasibility_eq_rhs_0 = infty_norm(qpwork.primal_residual_eq_scaled);
165 ruiz.unscale_primal_residual_in_place_in(
166 VectorViewMut<T>{ from_eigen, qpwork.primal_residual_in_scaled_up });
167 primal_feasibility_in_rhs_0 = infty_norm(qpwork.primal_residual_in_scaled_up);
168
169 qpwork.primal_residual_in_scaled_low =
170 helpers::positive_part(qpwork.primal_residual_in_scaled_up - qpmodel.u) +
171 helpers::negative_part(qpwork.primal_residual_in_scaled_up - qpmodel.l);
172 qpwork.primal_residual_eq_scaled -= qpmodel.b;
173
174 primal_feasibility_in_lhs = infty_norm(qpwork.primal_residual_in_scaled_low);
175 primal_feasibility_eq_lhs = infty_norm(qpwork.primal_residual_eq_scaled);
176 primal_feasibility_lhs =
177 std::max(primal_feasibility_eq_lhs, primal_feasibility_in_lhs);
178
179 ruiz.scale_primal_residual_in_place_eq(
180 VectorViewMut<T>{ from_eigen, qpwork.primal_residual_eq_scaled });
181}

198template<typename T>
199bool
200global_primal_residual_infeasibility(
201 VectorViewMut<T> ATdy,
202 VectorViewMut<T> CTdz,
203 VectorViewMut<T> dy,
204 VectorViewMut<T> dz,
205 Workspace<T>& qpwork,
206 const Settings<T>& qpsettings,
207 const preconditioner::RuizEquilibration<T>& ruiz)
208{
209
210 // The problem is primal infeasible if the following four conditions hold:
211 //
212 // ||unscaled(A^Tdy)|| <= eps_p_inf ||unscaled(dy)||
213 // b^T dy <= -eps_p_inf ||unscaled(dy)||
214 // ||unscaled(C^Tdz)|| <= eps_p_inf ||unscaled(dz)||
215 // u^T [dz]_+ - l^T[-dz]_+ <= -eps_p_inf ||unscaled(dz)||
216 //
217 // the variables in entry are changed in place
218
219 bool res = infty_norm(dy.to_eigen()) != 0 && infty_norm(dz.to_eigen()) != 0;
220 if (!res) {
221 return res;
222 }
223 ruiz.unscale_dual_residual_in_place(ATdy);
224 ruiz.unscale_dual_residual_in_place(CTdz);
225 T eq_inf = dy.to_eigen().dot(qpwork.b_scaled);
226 T in_inf = helpers::positive_part(dz.to_eigen()).dot(qpwork.u_scaled) -
227 helpers::negative_part(dz.to_eigen()).dot(qpwork.l_scaled);
228 ruiz.unscale_dual_in_place_eq(dy);
229 ruiz.unscale_dual_in_place_in(dz);
230
231 T bound_y = qpsettings.eps_primal_inf * infty_norm(dy.to_eigen());
232 T bound_z = qpsettings.eps_primal_inf * infty_norm(dz.to_eigen());
233
234 res = infty_norm(ATdy.to_eigen()) <= bound_y && eq_inf <= -bound_y &&
235 infty_norm(CTdz.to_eigen()) <= bound_z && in_inf <= -bound_z;
236 return res;
237}

256template<typename T>
257bool
258global_dual_residual_infeasibility(
259 VectorViewMut<T> Adx,
260 VectorViewMut<T> Cdx,
261 VectorViewMut<T> Hdx,
262 VectorViewMut<T> dx,
263 Workspace<T>& qpwork,
264 const Settings<T>& qpsettings,
265 const Model<T>& qpmodel,
266 const preconditioner::RuizEquilibration<T>& ruiz)
267{
268
269 // The problem is dual infeasible the two following conditions hold:
270 //
271 // FIRST
272 // ||unscaled(Adx)|| <= eps_d_inf ||unscaled(dx)||
273 // unscaled(Cdx)_i \in [-eps_d_inf,eps_d_inf] ||unscaled(dx)|| if u_i and l_i
274 // are finite or >=
275 // -eps_d_inf||unscaled(dx)|| if u_i = +inf or <= eps_d_inf||unscaled(dx)|| if
276 // l_i = -inf
277 //
278 // SECOND
279 //
280 // ||unscaled(Hdx)|| <= c eps_d_inf * ||unscaled(dx)|| and q^Tdx <= -c
281 // eps_d_inf ||unscaled(dx)|| the variables in
282 // entry are changed in place
283 ruiz.unscale_dual_residual_in_place(Hdx);
284 ruiz.unscale_primal_residual_in_place_eq(Adx);
285 ruiz.unscale_primal_residual_in_place_in(Cdx);
286 T gdx = (dx.to_eigen()).dot(qpwork.g_scaled);
287 ruiz.unscale_primal_in_place(dx);
288
289 T bound = infty_norm(dx.to_eigen()) * qpsettings.eps_dual_inf;
290 T bound_neg = -bound;
291
292 bool first_cond = infty_norm(Adx.to_eigen()) <= bound;
293
294 for (i64 iter = 0; iter < qpmodel.n_in; ++iter) {
295 T Cdx_i = Cdx.to_eigen()[iter];
296 if (qpwork.u_scaled[iter] <= 1.E20 && qpwork.l_scaled[iter] >= -1.E20) {
297 first_cond = first_cond && Cdx_i <= bound && Cdx_i >= bound_neg;
298 } else if (qpwork.u_scaled[iter] > 1.E20) {
299 first_cond = first_cond && Cdx_i >= bound_neg;
300 } else if (qpwork.l_scaled[iter] < -1.E20) {
301 first_cond = first_cond && Cdx_i <= bound;
302 }
303 }
304
305 bound *= ruiz.c;
306 bound_neg *= ruiz.c;
307 bool second_cond_alt1 =
308 infty_norm(Hdx.to_eigen()) <= bound && gdx <= bound_neg;
309 bound_neg *= qpsettings.eps_dual_inf;
310
311 bool res = first_cond && second_cond_alt1 && infty_norm(dx.to_eigen()) != 0;
312 return res;
313}

331template<typename T>
332void
333global_dual_residual(Results<T>& qpresults,
334 Workspace<T>& qpwork,
335 const Model<T>& qpmodel,
336 const preconditioner::RuizEquilibration<T>& ruiz,
337 T& dual_feasibility_lhs,
338 T& dual_feasibility_rhs_0,
339 T& dual_feasibility_rhs_1,
340 T& dual_feasibility_rhs_3,
341 T& rhs_duality_gap,
342 T& duality_gap)
343{
344 // dual_feasibility_lhs = norm(dual_residual_scaled)
345 // dual_feasibility_rhs_0 = norm(unscaled(Hx))
346 // dual_feasibility_rhs_1 = norm(unscaled(ATy))
347 // dual_feasibility_rhs_3 = norm(unscaled(CTz))
348 //
349 // dual_residual_scaled = scaled(Hx + g + ATy + CTz)
350 const isize max_dim =
351 std::max(qpmodel.dim, std::max(qpmodel.n_eq, qpmodel.n_in));
352 const T sqrt_max_dim(std::sqrt(max_dim)); // for normalizing scalar products
353
354 qpwork.dual_residual_scaled = qpwork.g_scaled;
355 qpwork.CTz.noalias() =
356 qpwork.H_scaled.template selfadjointView<Eigen::Lower>() * qpresults.x;
357 qpwork.dual_residual_scaled += qpwork.CTz;
358 ruiz.unscale_dual_residual_in_place(
359 VectorViewMut<T>{ from_eigen, qpwork.CTz }); // contains unscaled Hx
360 dual_feasibility_rhs_0 = infty_norm(qpwork.CTz);
361
362 ruiz.unscale_primal_in_place(VectorViewMut<T>{ from_eigen, qpresults.x });
363 duality_gap = (qpmodel.g).dot(qpresults.x);
364 rhs_duality_gap = std::abs(duality_gap);
365 const T xHx = (qpwork.CTz).dot(qpresults.x);
366 duality_gap += xHx; // contains now xHx+g.Tx
367 rhs_duality_gap = std::max(rhs_duality_gap, std::abs(xHx));
368
369 ruiz.scale_primal_in_place(VectorViewMut<T>{ from_eigen, qpresults.x });
370
371 qpwork.CTz.noalias() = qpwork.A_scaled.transpose() * qpresults.y;
372 qpwork.dual_residual_scaled += qpwork.CTz;
373 ruiz.unscale_dual_residual_in_place(
374 VectorViewMut<T>{ from_eigen, qpwork.CTz });
375 dual_feasibility_rhs_1 = infty_norm(qpwork.CTz);
376
377 qpwork.CTz.noalias() = qpwork.C_scaled.transpose() * qpresults.z;
378 qpwork.dual_residual_scaled += qpwork.CTz;
379 ruiz.unscale_dual_residual_in_place(
380 VectorViewMut<T>{ from_eigen, qpwork.CTz });
381 dual_feasibility_rhs_3 = infty_norm(qpwork.CTz);
382
383 ruiz.unscale_dual_residual_in_place(
384 VectorViewMut<T>{ from_eigen, qpwork.dual_residual_scaled });
385
386 dual_feasibility_lhs = infty_norm(qpwork.dual_residual_scaled);
387
388 ruiz.scale_dual_residual_in_place(
389 VectorViewMut<T>{ from_eigen, qpwork.dual_residual_scaled });
390
391 ruiz.unscale_dual_in_place_eq(VectorViewMut<T>{ from_eigen, qpresults.y });
392 const T by = (qpmodel.b).dot(qpresults.y);
393 rhs_duality_gap = std::max(rhs_duality_gap, std::abs(by));
394 duality_gap += by;
395 ruiz.scale_dual_in_place_eq(VectorViewMut<T>{ from_eigen, qpresults.y });
396
397 ruiz.unscale_dual_in_place_in(VectorViewMut<T>{ from_eigen, qpresults.z });
398
399 const T zu =
400 helpers::select(qpwork.active_set_up, qpresults.z, 0)
401 .dot(helpers::at_most(qpmodel.u, helpers::infinite_bound<T>::value()));
402 rhs_duality_gap = std::max(rhs_duality_gap, std::abs(zu));
403 duality_gap += zu;
404
405 const T zl =
406 helpers::select(qpwork.active_set_low, qpresults.z, 0)
407 .dot(helpers::at_least(qpmodel.l, -helpers::infinite_bound<T>::value()));
408 rhs_duality_gap = std::max(rhs_duality_gap, std::abs(zl));
409 duality_gap += zl;
410
411 ruiz.scale_dual_in_place_in(VectorViewMut<T>{ from_eigen, qpresults.z });
412
413 duality_gap /= sqrt_max_dim; // in order to get an a-dimensional duality gap
414 rhs_duality_gap /= sqrt_max_dim;
415}
