147template<typename T>
148void
149global_primal_residual(const Model<T>& qpmodel,
150 Results<T>& qpresults,
151 Workspace<T>& qpwork,
152 preconditioner::RuizEquilibration<T>& ruiz,
153 T& primal_feasibility_lhs,
154 T& primal_feasibility_eq_rhs_0,
155 T& primal_feasibility_in_rhs_0,
156 T& primal_feasibility_eq_lhs,
157 T& primal_feasibility_in_lhs)
158{
159 // COMPUTES:
160 // primal_residual_eq_scaled = scaled(Ax - b)
161 //
162 // primal_feasibility_lhs = max(norm(unscaled(Ax - b)),
163 // norm(unscaled([Cx - u]+ + [Cx - l]-)))
164 // primal_feasibility_eq_rhs_0 = norm(unscaled(Ax))
165 // primal_feasibility_in_rhs_0 = norm(unscaled(Cx))
166 //
167 // MAY_ALIAS[primal_residual_in_scaled_u, primal_residual_in_scaled_l]
168 //
169 // INDETERMINATE:
170 // primal_residual_in_scaled_u = unscaled(Cx)
171 // primal_residual_in_scaled_l = unscaled([Cx - u]+ + [Cx - l]-)
172 qpwork.primal_residual_eq_scaled.noalias() = qpwork.A_scaled * qpresults.x;
173 qpwork.primal_residual_in_scaled_up.noalias() = qpwork.C_scaled * qpresults.x;
174
175 ruiz.unscale_primal_residual_in_place_eq(
176 VectorViewMut<T>{ from_eigen, qpwork.primal_residual_eq_scaled });
177 primal_feasibility_eq_rhs_0 = infty_norm(qpwork.primal_residual_eq_scaled);
178 ruiz.unscale_primal_residual_in_place_in(
179 VectorViewMut<T>{ from_eigen, qpwork.primal_residual_in_scaled_up });
180 primal_feasibility_in_rhs_0 = infty_norm(qpwork.primal_residual_in_scaled_up);
181
182 qpwork.primal_residual_in_scaled_low =
183 positive_part(qpwork.primal_residual_in_scaled_up - qpmodel.u) +
184 negative_part(qpwork.primal_residual_in_scaled_up - qpmodel.l);
185 qpwork.primal_residual_eq_scaled -= qpmodel.b;
186
187 primal_feasibility_in_lhs = infty_norm(qpwork.primal_residual_in_scaled_low);
188 primal_feasibility_eq_lhs = infty_norm(qpwork.primal_residual_eq_scaled);
189 primal_feasibility_lhs =
190 std::max(primal_feasibility_eq_lhs, primal_feasibility_in_lhs);
191
192 ruiz.scale_primal_residual_in_place_eq(
193 VectorViewMut<T>{ from_eigen, qpwork.primal_residual_eq_scaled });
194}

211template<typename T>
212bool
213global_primal_residual_infeasibility(VectorViewMut<T> ATdy,
214 VectorViewMut<T> CTdz,
215 VectorViewMut<T> dy,
216 VectorViewMut<T> dz,
217 Workspace<T>& qpwork,
218 const Settings<T>& qpsettings,
219 preconditioner::RuizEquilibration<T>& ruiz)
220{
221
222 // The problem is primal infeasible if the following four conditions hold:
223 //
224 // ||unscaled(A^Tdy)|| <= eps_p_inf ||unscaled(dy)||
225 // b^T dy <= -eps_p_inf ||unscaled(dy)||
226 // ||unscaled(C^Tdz)|| <= eps_p_inf ||unscaled(dz)||
227 // u^T [dz]_+ - l^T[-dz]_+ <= -eps_p_inf ||unscaled(dz)||
228 //
229 // the variables in entry are changed in place
230
231 bool res = infty_norm(dy.to_eigen()) != 0 && infty_norm(dz.to_eigen()) != 0;
232 if (!res) {
233 return res;
234 }
235 ruiz.unscale_dual_residual_in_place(ATdy);
236 ruiz.unscale_dual_residual_in_place(CTdz);
237 T eq_inf = dy.to_eigen().dot(qpwork.b_scaled);
238 T in_inf = positive_part(dz.to_eigen()).dot(qpwork.u_scaled) -
239 positive_part(-dz.to_eigen()).dot(qpwork.l_scaled);
240 ruiz.unscale_dual_in_place_eq(dy);
241 ruiz.unscale_dual_in_place_in(dz);
242
243 T bound_y = qpsettings.eps_primal_inf * infty_norm(dy.to_eigen());
244 T bound_z = qpsettings.eps_primal_inf * infty_norm(dz.to_eigen());
245
246 res = infty_norm(ATdy.to_eigen()) <= bound_y && eq_inf <= -bound_y &&
247 infty_norm(CTdz.to_eigen()) <= bound_z && in_inf <= -bound_z;
248 return res;
249}

268template<typename T>
269bool
270global_dual_residual_infeasibility(VectorViewMut<T> Adx,
271 VectorViewMut<T> Cdx,
272 VectorViewMut<T> Hdx,
273 VectorViewMut<T> dx,
274 Workspace<T>& qpwork,
275 const Settings<T>& qpsettings,
276 const Model<T>& qpmodel,
277 preconditioner::RuizEquilibration<T>& ruiz)
278{
279
280 // The problem is dual infeasible the two following conditions hold:
281 //
282 // FIRST
283 // ||unscaled(Adx)|| <= eps_d_inf ||unscaled(dx)||
284 // unscaled(Cdx)_i \in [-eps_d_inf,eps_d_inf] ||unscaled(dx)|| if u_i and l_i
285 // are finite or >=
286 // -eps_d_inf||unscaled(dx)|| if u_i = +inf or <= eps_d_inf||unscaled(dx)|| if
287 // l_i = -inf
288 //
289 // SECOND
290 //
291 // ||unscaled(Hdx)|| <= c eps_d_inf * ||unscaled(dx)|| and q^Tdx <= -c
292 // eps_d_inf ||unscaled(dx)|| the variables in
293 // entry are changed in place
294 ruiz.unscale_dual_residual_in_place(Hdx);
295 ruiz.unscale_primal_residual_in_place_eq(Adx);
296 ruiz.unscale_primal_residual_in_place_in(Cdx);
297 T gdx = (dx.to_eigen()).dot(qpwork.g_scaled);
298 ruiz.unscale_primal_in_place(dx);
299
300 T bound = infty_norm(dx.to_eigen()) * qpsettings.eps_dual_inf;
301 T bound_neg = -bound;
302
303 bool first_cond = infty_norm(Adx.to_eigen()) <= bound;
304
305 for (i64 iter = 0; iter < qpmodel.n_in; ++iter) {
306 T Cdx_i = Cdx.to_eigen()[iter];
307 if (qpwork.u_scaled[iter] <= 1.E20 && qpwork.l_scaled[iter] >= -1.E20) {
308 first_cond = first_cond && Cdx_i <= bound && Cdx_i >= bound_neg;
309 } else if (qpwork.u_scaled[iter] > 1.E20) {
310 first_cond = first_cond && Cdx_i >= bound_neg;
311 } else if (qpwork.l_scaled[iter] < -1.E20) {
312 first_cond = first_cond && Cdx_i <= bound;
313 }
314 }
315
316 bound *= ruiz.c;
317 bound_neg *= ruiz.c;
318 bool second_cond_alt1 =
319 infty_norm(Hdx.to_eigen()) <= bound && gdx <= bound_neg;
320 bound_neg *= qpsettings.eps_dual_inf;
321
322 bool res = first_cond && second_cond_alt1 && infty_norm(dx.to_eigen()) != 0;
323 return res;
324}

342template<typename T>
343void
344global_dual_residual(Results<T>& qpresults,
345 Workspace<T>& qpwork,
346 preconditioner::RuizEquilibration<T>& ruiz,
347 T& dual_feasibility_lhs,
348 T& dual_feasibility_rhs_0,
349 T& dual_feasibility_rhs_1,
350 T& dual_feasibility_rhs_3)
351{
352 // dual_feasibility_lhs = norm(dual_residual_scaled)
353 // dual_feasibility_rhs_0 = norm(unscaled(Hx))
354 // dual_feasibility_rhs_1 = norm(unscaled(ATy))
355 // dual_feasibility_rhs_3 = norm(unscaled(CTz))
356 //
357 // dual_residual_scaled = scaled(Hx + g + ATy + CTz)
358 qpwork.dual_residual_scaled = qpwork.g_scaled;
359 qpwork.CTz.noalias() =
360 qpwork.H_scaled.template selfadjointView<Eigen::Lower>() * qpresults.x;
361 qpwork.dual_residual_scaled += qpwork.CTz;
362 ruiz.unscale_dual_residual_in_place(
363 VectorViewMut<T>{ from_eigen, qpwork.CTz });
364 dual_feasibility_rhs_0 = infty_norm(qpwork.CTz);
365 qpwork.CTz.noalias() = qpwork.A_scaled.transpose() * qpresults.y;
366 qpwork.dual_residual_scaled += qpwork.CTz;
367 ruiz.unscale_dual_residual_in_place(
368 VectorViewMut<T>{ from_eigen, qpwork.CTz });
369 dual_feasibility_rhs_1 = infty_norm(qpwork.CTz);
370
371 qpwork.CTz.noalias() = qpwork.C_scaled.transpose() * qpresults.z;
372 qpwork.dual_residual_scaled += qpwork.CTz;
373 ruiz.unscale_dual_residual_in_place(
374 VectorViewMut<T>{ from_eigen, qpwork.CTz });
375 dual_feasibility_rhs_3 = infty_norm(qpwork.CTz);
376
377 ruiz.unscale_dual_residual_in_place(
378 VectorViewMut<T>{ from_eigen, qpwork.dual_residual_scaled });
379
380 dual_feasibility_lhs = infty_norm(qpwork.dual_residual_scaled);
381
382 ruiz.scale_dual_residual_in_place(
383 VectorViewMut<T>{ from_eigen, qpwork.dual_residual_scaled });
384}
