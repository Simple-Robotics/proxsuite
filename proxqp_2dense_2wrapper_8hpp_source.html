80template<typename T>
81DenseBackend
82dense_backend_choice(DenseBackend _dense_backend,
83 isize dim,
84 isize n_eq,
85 isize n_in,
86 bool box_constraints)
87{
88 if (_dense_backend == DenseBackend::Automatic) {
89 isize n_constraints(n_in);
90 if (box_constraints) {
91 n_constraints += dim;
92 }
93 T threshold(1.5);
94 T frequence(0.2);
95 T PrimalDualLDLTCost =
96 0.5 * std::pow(T(n_eq) / T(dim), 2) +
97 0.17 * (std::pow(T(n_eq) / T(dim), 3) +
98 std::pow(T(n_constraints) / T(dim), 3)) +
99 frequence * std::pow(T(n_eq + n_constraints) / T(dim), 2) / T(dim);
100 T PrimalLDLTCost =
101 threshold *
102 ((0.5 * T(n_eq) + T(n_constraints)) / T(dim) + frequence / T(dim));
103 bool choice = PrimalDualLDLTCost > PrimalLDLTCost;
104 if (choice) {
105 return DenseBackend::PrimalLDLT;
106 } else {
107 return DenseBackend::PrimalDualLDLT;
108 }
109 } else {
110 return _dense_backend;
111 }
112}
