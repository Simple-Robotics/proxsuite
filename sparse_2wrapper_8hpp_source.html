606template<typename T, typename I>
607proxqp::Results<T>
608solve(
609 const SparseMat<T, I>& H,
610 const Vec<T>& g,
611 const SparseMat<T, I>& A,
612 const Vec<T>& b,
613 const SparseMat<T, I>& C,
614 const Vec<T>& u,
615 const Vec<T>& l,
616 std::optional<VecRef<T>> x = std::nullopt,
617 std::optional<VecRef<T>> y = std::nullopt,
618 std::optional<VecRef<T>> z = std::nullopt,
619 std::optional<T> eps_abs = std::nullopt,
620 std::optional<T> eps_rel = std::nullopt,
621 std::optional<T> rho = std::nullopt,
622 std::optional<T> mu_eq = std::nullopt,
623 std::optional<T> mu_in = std::nullopt,
624 std::optional<bool> verbose = std::nullopt,
625 bool compute_preconditioner = true,

627 std::optional<isize> max_iter = std::nullopt,

630{
631
632 isize n(H.rows());
633 isize n_eq(A.rows());
634 isize n_in(C.rows());
635
636 proxqp::sparse::QP<T, I> Qp(n, n_eq, n_in);
637 Qp.settings.initial_guess = initial_guess;
638
639 if (eps_abs != std::nullopt) {
640 Qp.settings.eps_abs = eps_abs.value();
641 }
642 if (eps_rel != std::nullopt) {
643 Qp.settings.eps_rel = eps_rel.value();
644 }
645 if (verbose != std::nullopt) {
646 Qp.settings.verbose = verbose.value();
647 }
648 if (max_iter != std::nullopt) {
649 Qp.settings.max_iter = verbose.value();
650 }
651 Qp.settings.compute_timings = compute_timings;
652 Qp.init(H, g, A, b, C, u, l, compute_preconditioner, rho, mu_eq, mu_in);
653 Qp.solve(x, y, z);
654
655 return Qp.results;
656}
