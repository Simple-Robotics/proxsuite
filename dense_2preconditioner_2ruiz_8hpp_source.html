30template<typename T>
31auto
32ruiz_scale_qp_in_place( //
33 VectorViewMut<T> delta_,
34 VectorViewMut<T> tmp_delta_preallocated,
35 std::ostream* logger_ptr,
36 QpViewBoxMut<T> qp,
37 T epsilon,
38 isize max_iter,
39 Symmetry sym) -> T
40{
41
42 T c(1);
43 auto S = delta_.to_eigen();
44
45 auto H = qp.H.to_eigen();
46 auto g = qp.g.to_eigen();
47 auto A = qp.A.to_eigen();
48 auto b = qp.b.to_eigen();
49 auto C = qp.C.to_eigen();
50 auto u = qp.u.to_eigen();
51 auto l = qp.l.to_eigen();
52
53 static constexpr T machine_eps = std::numeric_limits<T>::epsilon();
54 /*
55 * compute equilibration parameters and scale in place the qp following
56 * algorithm
57 *
58 * modified: removed g in gamma computation
59 */
60
61 isize n = qp.H.rows;
62 isize n_eq = qp.A.rows;
63 isize n_in = qp.C.rows;
64
65 T gamma = T(1);
66
67 auto delta = tmp_delta_preallocated.to_eigen();
68
69 i64 iter = 1;
70
71 while (infty_norm((1 - delta.array()).matrix()) > epsilon) {
72 if (logger_ptr != nullptr) {
73 *logger_ptr //
74 << "j : " //
75 << iter //
76 << " ; error : " //
77 << infty_norm((1 - delta.array()).matrix()) //
78 << "\n\n";
79 }
80 if (iter == max_iter) {
81 break;
82 } else {
83 ++iter;
84 }
85
86 // normalization vector
87 {
88 for (isize k = 0; k < n; ++k) {
89 switch (sym) {
90 case Symmetry::upper: { // upper triangular part
91 delta(k) = T(1) / (sqrt(std::max({
92 infty_norm(H.col(k).head(k)),
93 infty_norm(H.row(k).tail(n - k)),
94 infty_norm(A.col(k)),
95 infty_norm(C.col(k)),
96 })) +
97 machine_eps);
98 break;
99 }
100 case Symmetry::lower: { // lower triangular part
101 delta(k) = T(1) / (sqrt(std::max({
102 infty_norm(H.row(k).head(k)),
103 infty_norm(H.col(k).tail(n - k)),
104 infty_norm(A.col(k)),
105 infty_norm(C.col(k)),
106 })) +
107 machine_eps);
108 break;
109 }
110 case Symmetry::general: {
111 delta(k) = T(1) / (sqrt(std::max({
112 infty_norm(H.col(k)),
113 infty_norm(A.col(k)),
114 infty_norm(C.col(k)),
115 })) +
116 machine_eps);
117
118 break;
119 }
120 }
121 }
122
123 for (isize k = 0; k < n_eq; ++k) {
124 T aux = sqrt(infty_norm(A.row(k)));
125 delta(n + k) = T(1) / (aux + machine_eps);
126 }
127 for (isize k = 0; k < n_in; ++k) {
128 T aux = sqrt(infty_norm(C.row(k)));
129 delta(k + n + n_eq) = T(1) / (aux + machine_eps);
130 }
131 }
132 {
133
134 // normalize A and C
135 A = delta.segment(n, n_eq).asDiagonal() * A * delta.head(n).asDiagonal();
136 C = delta.tail(n_in).asDiagonal() * C * delta.head(n).asDiagonal();
137 // normalize vectors
138 g.array() *= delta.head(n).array();
139 b.array() *= delta.middleRows(n, n_eq).array();
140 u.array() *= delta.tail(n_in).array();
141 l.array() *= delta.tail(n_in).array();
142
143 // normalize H
144 switch (sym) {
145 case Symmetry::upper: {
146 // upper triangular part
147 for (isize j = 0; j < n; ++j) {
148 H.col(j).head(j + 1) *= delta(j);
149 }
150 // normalisation des lignes
151 for (isize i = 0; i < n; ++i) {
152 H.row(i).tail(n - i) *= delta(i);
153 }
154 break;
155 }
156 case Symmetry::lower: {
157 // lower triangular part
158 for (isize j = 0; j < n; ++j) {
159 H.col(j).tail(n - j) *= delta(j);
160 }
161 // normalisation des lignes
162 for (isize i = 0; i < n; ++i) {
163 H.row(i).head(i + 1) *= delta(i);
164 }
165 break;
166 }
167 case Symmetry::general: {
168 // all matrix
169 H = delta.head(n).asDiagonal() * H * delta.head(n).asDiagonal();
170 break;
171 }
172 default:
173 break;
174 }
175
176 // additional normalization for the cost function
177 switch (sym) {
178 case Symmetry::upper: {
179 // upper triangular part
180 T tmp = T(0);
181 for (isize j = 0; j < n; ++j) {
182 tmp += proxqp::dense::infty_norm(H.row(j).tail(n - j));
183 }
184 gamma = 1 / std::max(tmp / T(n), T(1));
185 break;
186 }
187 case Symmetry::lower: {
188 // lower triangular part
189 T tmp = T(0);
190 for (isize j = 0; j < n; ++j) {
191 tmp += proxqp::dense::infty_norm(H.col(j).tail(n - j));
192 }
193 gamma = 1 / std::max(tmp / T(n), T(1));
194 break;
195 }
196 case Symmetry::general: {
197 // all matrix
198 gamma =
199 1 /
200 std::max(T(1),
201 (H.colwise().template lpNorm<Eigen::Infinity>()).mean());
202 break;
203 }
204 default:
205 break;
206 }
207
208 g *= gamma;
209 H *= gamma;
210
211 S.array() *= delta.array(); // coefficientwise product
212 c *= gamma;
213 }
214 }
215 return c;
216}

302

306 delta.setOnes();

308 c = detail::ruiz_scale_qp_in_place({ proxqp::from_eigen, delta },
309 { proxqp::from_eigen, tmp_delta },
310 logger_ptr,
311 qp,
312 epsilon,
313 max_iter,
314 sym);
315 } else {
316
317 auto H = qp.H.to_eigen();
318 auto g = qp.g.to_eigen();
319 auto A = qp.A.to_eigen();
320 auto b = qp.b.to_eigen();
321 auto C = qp.C.to_eigen();
322 auto u = qp.u.to_eigen();
323 auto l = qp.l.to_eigen();

330 C = delta.tail(n_in).asDiagonal() * C * delta.head(n).asDiagonal();
331

334 case Symmetry::upper: {
335 // upper triangular part
336 for (isize j = 0; j < n; ++j) {
337 H.col(j).head(j + 1) *= delta(j);

339 // normalisation des lignes

341 H.row(i).tail(n - i) *= delta(i);

359 break;

364

367 b.array() *= delta.segment(n, n_eq).array();
368 l.array() *= delta.tail(n_in).array();
369 u.array() *= delta.tail(n_in).array();

372 H *= c;
