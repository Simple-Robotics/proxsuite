28template<typename T>
29auto
30ruiz_scale_qp_in_place( //
31 VectorViewMut<T> delta_,
32 std::ostream* logger_ptr,
33 QpViewBoxMut<T> qp,
34 T epsilon,
35 isize max_iter,
36 Symmetry sym,
37 proxsuite::linalg::veg::dynstack::DynStackMut stack) -> T
38{
39
40 T c(1);
41 auto S = delta_.to_eigen();
42
43 auto H = qp.H.to_eigen();
44 auto g = qp.g.to_eigen();
45 auto A = qp.A.to_eigen();
46 auto b = qp.b.to_eigen();
47 auto C = qp.C.to_eigen();
48 auto u = qp.u.to_eigen();
49 auto l = qp.l.to_eigen();
50
51 static constexpr T machine_eps = std::numeric_limits<T>::epsilon();
52 /*
53 * compute equilibration parameters and scale in place the qp following
54 * algorithm
55 *
56 * modified: removed g in gamma computation
57 */
58
59 isize n = qp.H.rows;
60 isize n_eq = qp.A.rows;
61 isize n_in = qp.C.rows;
62
63 T gamma = T(1);
64
65 LDLT_TEMP_VEC(T, delta, n + n_eq + n_in, stack);
66
67 i64 iter = 1;
68
69 while (infty_norm((1 - delta.array()).matrix()) > epsilon) {
70 if (logger_ptr != nullptr) {
71 *logger_ptr //
72 << "j : " //
73 << iter //
74 << " ; error : " //
75 << infty_norm((1 - delta.array()).matrix()) //
76 << "\n\n";
77 }
78 if (iter == max_iter) {
79 break;
80 } else {
81 ++iter;
82 }
83
84 // normalization vector
85 {
86 for (isize k = 0; k < n; ++k) {
87 switch (sym) {
88 case Symmetry::upper: { // upper triangular part
89 T aux = sqrt(std::max({
90 infty_norm(H.col(k).head(k)),
91 infty_norm(H.row(k).tail(n - k)),
92 n_eq > 0 ? infty_norm(A.col(k)) : T(0),
93 n_in > 0 ? infty_norm(C.col(k)) : T(0),
94 }));
95 if (aux == T(0)) {
96 delta(k) = T(1);
97 } else {
98 delta(k) = T(1) / (aux + machine_eps);
99 }
100 break;
101 }
102 case Symmetry::lower: { // lower triangular part
103
104 T aux = sqrt(std::max({
105 infty_norm(H.col(k).head(k)),
106 infty_norm(H.col(k).tail(n - k)),
107 n_eq > 0 ? infty_norm(A.col(k)) : T(0),
108 n_in > 0 ? infty_norm(C.col(k)) : T(0),
109 }));
110 if (aux == T(0)) {
111 delta(k) = T(1);
112 } else {
113 delta(k) = T(1) / (aux + machine_eps);
114 }
115 break;
116 }
117 case Symmetry::general: {
118
119 T aux = sqrt(std::max({
120 infty_norm(H.col(k)),
121 n_eq > 0 ? infty_norm(A.col(k)) : T(0),
122 n_in > 0 ? infty_norm(C.col(k)) : T(0),
123 }));
124 if (aux == T(0)) {
125 delta(k) = T(1);
126 } else {
127 delta(k) = T(1) / (aux + machine_eps);
128 }
129 break;
130 }
131 }
132 }
133
134 for (isize k = 0; k < n_eq; ++k) {
135 T aux = sqrt(infty_norm(A.row(k)));
136 if (aux == T(0)) {
137 delta(n + k) = T(1);
138 } else {
139 delta(n + k) = T(1) / (aux + machine_eps);
140 }
141 }
142 for (isize k = 0; k < n_in; ++k) {
143 T aux = sqrt(infty_norm(C.row(k)));
144 if (aux == T(0)) {
145 delta(k + n + n_eq) = T(1);
146 } else {
147 delta(k + n + n_eq) = T(1) / (aux + machine_eps);
148 }
149 }
150 }
151 {
152
153 // normalize A and C
154 A = delta.segment(n, n_eq).asDiagonal() * A * delta.head(n).asDiagonal();
155 C = delta.tail(n_in).asDiagonal() * C * delta.head(n).asDiagonal();
156 // normalize vectors
157 g.array() *= delta.head(n).array();
158 b.array() *= delta.middleRows(n, n_eq).array();
159 u.array() *= delta.tail(n_in).array();
160 l.array() *= delta.tail(n_in).array();
161
162 // normalize H
163 switch (sym) {
164 case Symmetry::upper: {
165 // upper triangular part
166 for (isize j = 0; j < n; ++j) {
167 H.col(j).head(j + 1) *= delta(j);
168 }
169 // normalisation des lignes
170 for (isize i = 0; i < n; ++i) {
171 H.row(i).tail(n - i) *= delta(i);
172 }
173 break;
174 }
175 case Symmetry::lower: {
176 // lower triangular part
177 for (isize j = 0; j < n; ++j) {
178 H.col(j).tail(n - j) *= delta(j);
179 }
180 // normalisation des lignes
181 for (isize i = 0; i < n; ++i) {
182 H.row(i).head(i + 1) *= delta(i);
183 }
184 break;
185 }
186 case Symmetry::general: {
187 // all matrix
188 H = delta.head(n).asDiagonal() * H * delta.head(n).asDiagonal();
189 break;
190 }
191 default:
192 break;
193 }
194
195 // additional normalization for the cost function
196 switch (sym) {
197 case Symmetry::upper: {
198 // upper triangular part
199 T tmp = T(0);
200 for (isize j = 0; j < n; ++j) {
201 tmp += proxqp::dense::infty_norm(H.row(j).tail(n - j));
202 }
203 gamma = 1 / std::max(tmp / T(n), T(1));
204 break;
205 }
206 case Symmetry::lower: {
207 // lower triangular part
208 T tmp = T(0);
209 for (isize j = 0; j < n; ++j) {
210 tmp += proxqp::dense::infty_norm(H.col(j).tail(n - j));
211 }
212 gamma = 1 / std::max(tmp / T(n), T(1));
213 break;
214 }
215 case Symmetry::general: {
216 // all matrix
217 gamma =
218 1 /
219 std::max(T(1),
220 (H.colwise().template lpNorm<Eigen::Infinity>()).mean());
221 break;
222 }
223 default:
224 break;
225 }
226
227 g *= gamma;
228 H *= gamma;
229
230 S.array() *= delta.array(); // coefficientwise product
231 c *= gamma;
232 }
233 }
234 return c;
235}

301 // H_new = c * head @ H @ head
302 // A_new = tail @ A @ head
303 // g_new = c * head @ g
304 // b_new = tail @ b
315 void scale_qp_in_place(QpViewBoxMut<T> qp,
316 bool execute_preconditioner,
317 const isize max_iter,
318 const T epsilon,
319 proxsuite::linalg::veg::dynstack::DynStackMut stack)
320 {
321 if (execute_preconditioner) {
322 delta.setOnes();
323 c = detail::ruiz_scale_qp_in_place({ proxqp::from_eigen, delta },
324 logger_ptr,
325 qp,
326 epsilon,
327 max_iter,
328 sym,
329 stack);
330 } else {
331
332 auto H = qp.H.to_eigen();
333 auto g = qp.g.to_eigen();
334 auto A = qp.A.to_eigen();
335 auto b = qp.b.to_eigen();
336 auto C = qp.C.to_eigen();
337 auto u = qp.u.to_eigen();
338 auto l = qp.l.to_eigen();
339 isize n = qp.H.rows;
340 isize n_eq = qp.A.rows;
341 isize n_in = qp.C.rows;
342
343 // normalize A and C
344 A = delta.segment(n, n_eq).asDiagonal() * A * delta.head(n).asDiagonal();
345 C = delta.tail(n_in).asDiagonal() * C * delta.head(n).asDiagonal();
346
347 // normalize H
348 switch (sym) {
349 case Symmetry::upper: {
350 // upper triangular part
351 for (isize j = 0; j < n; ++j) {
352 H.col(j).head(j + 1) *= delta(j);
353 }
354 // normalisation des lignes
355 for (isize i = 0; i < n; ++i) {
356 H.row(i).tail(n - i) *= delta(i);
357 }
358 break;
359 }
360 case Symmetry::lower: {
361 // lower triangular part
362 for (isize j = 0; j < n; ++j) {
363 H.col(j).tail(n - j) *= delta(j);
364 }
365 // normalisation des lignes
366 for (isize i = 0; i < n; ++i) {
367 H.row(i).head(i + 1) *= delta(i);
368 }
369 break;
370 }
371 case Symmetry::general: {
372 // all matrix
373 H = delta.head(n).asDiagonal() * H * delta.head(n).asDiagonal();
374 break;
375 }
376 default:
377 break;
378 }
379
380 // normalize vectors
381 g.array() *= delta.head(n).array();
382 b.array() *= delta.segment(n, n_eq).array();
383 l.array() *= delta.tail(n_in).array();
384 u.array() *= delta.tail(n_in).array();
385
386 g *= c;
387 H *= c;
388 }

398 void scale_qp(const QpViewBox<T> qp,
399 QpViewBoxMut<T> scaled_qp,
400 VectorViewMut<T> tmp_delta_preallocated) const
401 {
402
403 /*
404 * scaled_qp is scaled, whereas first qp is not
405 * the procedure computes as well equilibration parameters using default
406 * parameters
407 */
408
409 scaled_qp.H.to_eigen() = qp.H.to_eigen();
410 scaled_qp.A.to_eigen() = qp.A.to_eigen();
411 scaled_qp.C.to_eigen() = qp.C.to_eigen();
412 scaled_qp.g.to_eigen() = qp.g.to_eigen();
413 scaled_qp.b.to_eigen() = qp.b.to_eigen();
414 scaled_qp.d.to_eigen() = qp.d.to_eigen();
415
416 scale_qp_in_place(scaled_qp, tmp_delta_preallocated, epsilon, max_iter);
417 }
