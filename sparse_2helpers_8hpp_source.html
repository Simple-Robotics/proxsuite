26template<typename T, typename I>
27void
28update_proximal_parameters(Results<T>& results,
29 Workspace<T, I>& work,
30 std::optional<T> rho_new,
31 std::optional<T> mu_eq_new,
32 std::optional<T> mu_in_new)
33{
34 if (rho_new != std::nullopt) {
35 results.info.rho = rho_new.value();
36 work.internal.proximal_parameter_update = true;
37 }
38 if (mu_eq_new != std::nullopt) {
39 results.info.mu_eq = mu_eq_new.value();
40 results.info.mu_eq_inv = T(1) / results.info.mu_eq;
41 work.internal.proximal_parameter_update = true;
42 }
43 if (mu_in_new != std::nullopt) {
44 results.info.mu_in = mu_in_new.value();
45 results.info.mu_in_inv = T(1) / results.info.mu_in;
46 work.internal.proximal_parameter_update = true;
47 }
48}
58template<typename T, typename I>
59void
60warm_start(std::optional<VecRef<T>> x_wm,
61 std::optional<VecRef<T>> y_wm,
62 std::optional<VecRef<T>> z_wm,
63 Results<T>& results,
64 Settings<T>& settings,
65 Model<T, I>& model)
66{
67
68 isize n_eq = results.y.rows();
69 isize n_in = results.z.rows();
70 if (n_eq != 0) {
71 if (n_in != 0) {
72 if (x_wm != std::nullopt && y_wm != std::nullopt &&
73 z_wm != std::nullopt) {
74 PROXSUITE_CHECK_ARGUMENT_SIZE(
75 z_wm.value().rows(),
76 model.n_in,
77 "the dimension wrt inequality constrained variables for warm start "
78 "is not valid.");
79 PROXSUITE_CHECK_ARGUMENT_SIZE(y_wm.value().rows(),
80 model.n_eq,
81 "the dimension wrt equality constrained "
82 "variables for warm start is not valid.");
83 PROXSUITE_CHECK_ARGUMENT_SIZE(x_wm.value().rows(),
84 model.dim,
85 "the dimension wrt primal variable x "
86 "variable for warm start is not valid.");
87 results.x = x_wm.value().eval();
88 results.y = y_wm.value().eval();
89 results.z = z_wm.value().eval();
90 settings.initial_guess = InitialGuessStatus::WARM_START;
91 }
92 } else {
93 // n_in= 0
94 if (x_wm != std::nullopt && y_wm != std::nullopt) {
95 PROXSUITE_CHECK_ARGUMENT_SIZE(y_wm.value().rows(),
96 model.n_eq,
97 "the dimension wrt equality constrained "
98 "variables for warm start is not valid.");
99 PROXSUITE_CHECK_ARGUMENT_SIZE(x_wm.value().rows(),
100 model.dim,
101 "the dimension wrt primal variable x "
102 "variable for warm start is not valid.");
103 results.x = x_wm.value().eval();
104 results.y = y_wm.value().eval();
105 settings.initial_guess = InitialGuessStatus::WARM_START;
106 }
107 }
108 } else if (n_in != 0) {
109 // n_eq = 0
110 if (x_wm != std::nullopt && z_wm != std::nullopt) {
111 PROXSUITE_CHECK_ARGUMENT_SIZE(z_wm.value().rows(),
112 model.n_in,
113 "the dimension wrt inequality constrained "
114 "variables for warm start is not valid.");
115 PROXSUITE_CHECK_ARGUMENT_SIZE(x_wm.value().rows(),
116 model.dim,
117 "the dimension wrt primal variable x "
118 "variable for warm start is not valid.");
119 results.x = x_wm.value().eval();
120 results.z = z_wm.value().eval();
121 settings.initial_guess = InitialGuessStatus::WARM_START;
122 }
123 } else {
124 // n_eq = 0 and n_in = 0
125 if (x_wm != std::nullopt) {
126 PROXSUITE_CHECK_ARGUMENT_SIZE(x_wm.value().rows(),
127 model.dim,
128 "the dimension wrt primal variable x "
129 "variable for warm start is not valid.");
130 results.x = x_wm.value().eval();
131 settings.initial_guess = InitialGuessStatus::WARM_START;
132 }
133 }

149template<typename T, typename I, typename P>
150void
151qp_setup(QpView<T, I> qp,
152 Results<T>& results,
153 Model<T, I>& data,
154 Workspace<T, I>& work,
155 Settings<T>& settings,
156 P& precond,
157 PreconditionerStatus& preconditioner_status)
158{
159 isize n = qp.H.nrows();
160 isize n_eq = qp.AT.ncols();
161 isize n_in = qp.CT.ncols();
162
163 if (results.x.rows() != n) {
164 results.x.resize(n);
165 results.x.setZero();
166 }
167 if (results.y.rows() != n_eq) {
168 results.y.resize(n_eq);
169 results.y.setZero();
170 }
171 if (results.z.rows() != n_in) {
172 results.z.resize(n_in);
173 results.z.setZero();
174 }
175 if (results.active_constraints.len() != n_in) {
176 results.active_constraints.resize(n_in);
177 for (isize i = 0; i < n_in; ++i) {
178 results.active_constraints[i] = false;
179 }
180 }
181 bool execute_preconditioner_or_not = false;
182 switch (preconditioner_status) {
183 case PreconditionerStatus::EXECUTE:
184 execute_preconditioner_or_not = true;
185 break;
186 case PreconditionerStatus::IDENTITY:
187 execute_preconditioner_or_not = false;
188 break;
189 case PreconditionerStatus::KEEP:
190 // keep previous one
191 execute_preconditioner_or_not = false;
192 break;
193 }
194 // performs scaling according to options chosen + stored model value
195 work.setup_impl(
196 qp,
197 data,
198 settings,
199 execute_preconditioner_or_not,
200 precond,
201 P::scale_qp_in_place_req(proxsuite::linalg::veg::Tag<T>{}, n, n_eq, n_in));
202 switch (settings.initial_guess) { // the following is used when initiliazing
203 // the Qp object or updating it
204 case InitialGuessStatus::EQUALITY_CONSTRAINED_INITIAL_GUESS: {
205
206 if (work.internal.proximal_parameter_update) {
207 results.cleanup_all_except_prox_parameters();
208 } else {
209 results.cleanup();
210 }
211 break;
212 }
213 case InitialGuessStatus::COLD_START_WITH_PREVIOUS_RESULT: {
214 // keep solutions but restart workspace and results
215
216 if (work.internal.proximal_parameter_update) {
217 results.cleanup_statistics();
218 } else {
219 results.cold_start();
220 }
221 break;
222 }
223 case InitialGuessStatus::NO_INITIAL_GUESS: {
224
225 if (work.internal.proximal_parameter_update) {
226 results.cleanup_all_except_prox_parameters();
227 } else {
228 results.cleanup();
229 }
230 break;
231 }
232 case InitialGuessStatus::WARM_START: {
233
234 if (work.internal.proximal_parameter_update) {
235 results.cleanup_all_except_prox_parameters();
236 } else {
237 results.cleanup();
238 }
239 break;
240 }
241 case InitialGuessStatus::WARM_START_WITH_PREVIOUS_RESULT: {
242 // keep workspace and results solutions except statistics
243
244 results.cleanup_statistics(); // always keep prox parameters (changed or
245 // previous ones)
246 break;
247 }
248 }

256template<typename T, typename I>
257auto
258have_same_structure(proxsuite::linalg::sparse::MatRef<T, I> a,
259 proxsuite::linalg::sparse::MatRef<T, I> b) -> bool
260{
261 if (a.nrows() != b.nrows())
262 return false;
263 if (a.ncols() != b.ncols())
264 return false;
265 for (usize j = 0; j < static_cast<usize>(a.ncols()); ++j) {
266 usize n_elems(a.col_end(j) - a.col_start(j));
267 usize n_elems_to_compare(b.col_end(j) - b.col_start(j));
268 if (n_elems != n_elems_to_compare)
269 return false;
270 for (usize p = 0; p < n_elems; ++p) {
271 isize i_a = a.row_indices()[a.col_start(j) + p];
272 isize i_b = b.row_indices()[b.col_start(j) + p];
273 if (i_a != i_b)
274 return false;
275 }
276 }
277 return true;
278}
285template<typename T, typename I>
286void
287copy(proxsuite::linalg::sparse::MatMut<T, I> a,
288 proxsuite::linalg::sparse::MatRef<T, I> b)
289{
290 // assume same sparsity structure for a and b
291 // copy b into a
292 for (usize j = 0; j < static_cast<usize>(a.ncols()); ++j) {
293 auto a_start = a.values_mut() + a.col_start(j);
294 auto b_start = b.values() + b.col_start(j);
295
296 usize n_elems = static_cast<usize>(a.col_end(j) - a.col_start(j));
297
298 for (usize p = 0; p < n_elems; ++p) {
299 a_start[p] = b_start[p];
300 }
301 }
302}
