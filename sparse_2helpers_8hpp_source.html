26template<typename T, typename I>
27void
28update_proximal_parameters(Settings<T>& settings,
29 Results<T>& results,
30 Workspace<T, I>& work,
31 optional<T> rho_new,
32 optional<T> mu_eq_new,
33 optional<T> mu_in_new)
34{
35 if (rho_new != nullopt) {
36 settings.default_rho = rho_new.value();
37 results.info.rho = rho_new.value();
38 work.internal.proximal_parameter_update = true;
39 }
40 if (mu_eq_new != nullopt) {
41 settings.default_mu_eq = mu_eq_new.value();
42 results.info.mu_eq = mu_eq_new.value();
43 results.info.mu_eq_inv = T(1) / results.info.mu_eq;
44 work.internal.proximal_parameter_update = true;
45 }
46 if (mu_in_new != nullopt) {
47 settings.default_mu_in = mu_in_new.value();
48 results.info.mu_in = mu_in_new.value();
49 results.info.mu_in_inv = T(1) / results.info.mu_in;
50 work.internal.proximal_parameter_update = true;
51 }
52}
62template<typename T, typename I>
63void
64warm_start(optional<VecRef<T>> x_wm,
65 optional<VecRef<T>> y_wm,
66 optional<VecRef<T>> z_wm,
67 Results<T>& results,
68 Settings<T>& settings,
69 Model<T, I>& model)
70{
71 if (x_wm == nullopt && y_wm == nullopt && z_wm == nullopt)
72 return;
73
74 settings.initial_guess = InitialGuessStatus::WARM_START;
75
76 // first check problem dimensions
77 if (x_wm != nullopt) {
78 PROXSUITE_CHECK_ARGUMENT_SIZE(
79 x_wm.value().rows(),
80 model.dim,
81 "the dimension wrt primal variable x for warm start is not valid.");
82 }
83
84 if (y_wm != nullopt) {
85 PROXSUITE_CHECK_ARGUMENT_SIZE(y_wm.value().rows(),
86 model.n_eq,
87 "the dimension wrt equality constrained "
88 "variables for warm start is not valid.");
89 }
90
91 if (z_wm != nullopt) {
92 PROXSUITE_CHECK_ARGUMENT_SIZE(
93 z_wm.value().rows(),
94 model.n_in,
95 "the dimension wrt inequality constrained variables for warm start "
96 "is not valid.");
97 }
98
99 if (x_wm != nullopt) {
100 results.x = x_wm.value().eval();
101 }
102
103 if (y_wm != nullopt) {
104 results.y = y_wm.value().eval();
105 }
106
107 if (z_wm != nullopt) {
108 results.z = z_wm.value().eval();
109 }
110}

125template<typename T, typename I, typename P>
126void
127qp_setup(QpView<T, I> qp,
128 Results<T>& results,
129 Model<T, I>& data,
130 Workspace<T, I>& work,
131 Settings<T>& settings,
132 P& precond,
133 PreconditionerStatus& preconditioner_status)
134{
135 isize n = qp.H.nrows();
136 isize n_eq = qp.AT.ncols();
137 isize n_in = qp.CT.ncols();
138
139 if (results.x.rows() != n) {
140 results.x.resize(n);
141 results.x.setZero();
142 }
143 if (results.y.rows() != n_eq) {
144 results.y.resize(n_eq);
145 results.y.setZero();
146 }
147 if (results.z.rows() != n_in) {
148 results.z.resize(n_in);
149 results.z.setZero();
150 }
151 if (results.active_constraints.len() != n_in) {
152 results.active_constraints.resize(n_in);
153 for (isize i = 0; i < n_in; ++i) {
154 results.active_constraints[i] = false;
155 }
156 }
157 bool execute_preconditioner_or_not = false;
158 switch (preconditioner_status) {
159 case PreconditionerStatus::EXECUTE:
160 execute_preconditioner_or_not = true;
161 break;
162 case PreconditionerStatus::IDENTITY:
163 execute_preconditioner_or_not = false;
164 break;
165 case PreconditionerStatus::KEEP:
166 // keep previous one
167 execute_preconditioner_or_not = false;
168 break;
169 }
170 // performs scaling according to options chosen + stored model value
171 work.setup_impl(
172 qp,
173 data,
174 settings,
175 execute_preconditioner_or_not,
176 precond,
177 P::scale_qp_in_place_req(proxsuite::linalg::veg::Tag<T>{}, n, n_eq, n_in));
178 switch (settings.initial_guess) { // the following is used when initiliazing
179 // the Qp object or updating it
180 case InitialGuessStatus::EQUALITY_CONSTRAINED_INITIAL_GUESS: {
181
182 if (work.internal.proximal_parameter_update) {
183 results.cleanup_all_except_prox_parameters();
184 } else {
185 results.cleanup(settings);
186 }
187 break;
188 }
189 case InitialGuessStatus::COLD_START_WITH_PREVIOUS_RESULT: {
190 // keep solutions but restart workspace and results
191
192 if (work.internal.proximal_parameter_update) {
193 results.cleanup_statistics();
194 } else {
195 results.cold_start(settings);
196 }
197 break;
198 }
199 case InitialGuessStatus::NO_INITIAL_GUESS: {
200
201 if (work.internal.proximal_parameter_update) {
202 results.cleanup_all_except_prox_parameters();
203 } else {
204 results.cleanup(settings);
205 }
206 break;
207 }
208 case InitialGuessStatus::WARM_START: {
209
210 if (work.internal.proximal_parameter_update) {
211 results.cleanup_all_except_prox_parameters();
212 } else {
213 results.cleanup(settings);
214 }
215 break;
216 }
217 case InitialGuessStatus::WARM_START_WITH_PREVIOUS_RESULT: {
218 // keep workspace and results solutions except statistics
219
220 results.cleanup_statistics(); // always keep prox parameters (changed or
221 // previous ones)
222 break;
223 }
224 }
225 // if user chose Automatic as sparse backend, store in results which backend
226 // of SparseCholesky or MatrixFree had been used
227 if (settings.sparse_backend == SparseBackend::Automatic) {
228 if (work.internal.do_ldlt) {
229 results.info.sparse_backend = SparseBackend::SparseCholesky;
230 } else {
231 results.info.sparse_backend = SparseBackend::MatrixFree;
232 }
233 }
234 // if user selected a specfic sparse backend, store it in results
235 else {
236 results.info.sparse_backend = settings.sparse_backend;
237 }
238}
245template<typename T, typename I>
246auto
247have_same_structure(proxsuite::linalg::sparse::MatRef<T, I> a,
248 proxsuite::linalg::sparse::MatRef<T, I> b) -> bool
249{
250 if (a.nrows() != b.nrows())
251 return false;
252 if (a.ncols() != b.ncols())
253 return false;
254 for (usize j = 0; j < static_cast<usize>(a.ncols()); ++j) {
255 usize n_elems(a.col_end(j) - a.col_start(j));
256 usize n_elems_to_compare(b.col_end(j) - b.col_start(j));
257 if (n_elems != n_elems_to_compare)
258 return false;
259 for (usize p = 0; p < n_elems; ++p) {
260 isize i_a = a.row_indices()[a.col_start(j) + p];
261 isize i_b = b.row_indices()[b.col_start(j) + p];
262 if (i_a != i_b)
263 return false;
264 }
265 }
266 return true;
267}
274template<typename T, typename I>
275void
276copy(proxsuite::linalg::sparse::MatMut<T, I> a,
277 proxsuite::linalg::sparse::MatRef<T, I> b)
278{
279 // assume same sparsity structure for a and b
280 // copy b into a
281 for (usize j = 0; j < static_cast<usize>(a.ncols()); ++j) {
282 auto a_start = a.values_mut() + a.col_start(j);
283 auto b_start = b.values() + b.col_start(j);
284
285 usize n_elems = static_cast<usize>(a.col_end(j) - a.col_start(j));
286
287 for (usize p = 0; p < n_elems; ++p) {
288 a_start[p] = b_start[p];
289 }
290 }
291}
