50template<typename T>
51auto
52primal_dual_derivative_results(const Model<T>& qpmodel,
53 Results<T>& qpresults,
54 Workspace<T>& qpwork,
55 T alpha) -> PrimalDualDerivativeResult<T>
56{
57
58 /*
59 * the function computes the first derivative of phi(alpha) at outer step k
60 * and inner step l
61 *
62 * phi(alpha) = f(x_l+alpha dx) + rho/2 |x_l + alpha dx - x_k|**2
63 * + mu_eq_inv/2 (|A(x_l+alpha dx)-d+y_k * mu_eq|**2)
64 * + mu_eq_inv * nu /2 (|A(x_l+alpha dx)-d+y_k * mu_eq -
65 * (y_l+alpha dy)
66 * |**2)
67 * + mu_in_inv/2 ( | [C(x_l+alpha dx) - u + z_k * mu_in]_+ |**2
68 * +| [C(x_l+alpha dx) - l + z_k * mu_in]_- |**2
69 * )
70 * + mu_in_inv * nu / 2 ( | [C(x_l+alpha dx) - u +
71 * z_k
72 * * mu_in]_+
73 * + [C(x_l+alpha dx) - l + z_k * mu_in]_- - (z+alpha dz) * mu_in |**2 with
74 * f(x) = 0.5 * x^THx + g^Tx phi is a second order polynomial in alpha. Below
75 * are computed its coefficients a0 and b0 in order to compute the desired
76 * gradient a0 * alpha + b0
77 */
78
79 qpwork.primal_residual_in_scaled_up_plus_alphaCdx =
80 qpwork.primal_residual_in_scaled_up + qpwork.Cdx * alpha;
81 qpwork.primal_residual_in_scaled_low_plus_alphaCdx =
82 qpwork.primal_residual_in_scaled_low + qpwork.Cdx * alpha;
83
84 T a(qpwork.dw_aug.head(qpmodel.dim).dot(qpwork.Hdx) +
85 qpresults.info.mu_eq_inv * (qpwork.Adx).squaredNorm() +
86 qpresults.info.rho *
87 qpwork.dw_aug.head(qpmodel.dim)
88 .squaredNorm()); // contains now: a = dx.dot(H.dot(dx)) + rho *
89 // norm(dx)**2 + (mu_eq_inv) * norm(Adx)**2
90
91 qpwork.err.segment(qpmodel.dim, qpmodel.n_eq) =
92 qpwork.Adx -
93 qpwork.dw_aug.segment(qpmodel.dim, qpmodel.n_eq) * qpresults.info.mu_eq;
94 a += qpwork.err.segment(qpmodel.dim, qpmodel.n_eq).squaredNorm() *
95 qpresults.info.mu_eq_inv *
96 qpresults.info
97 .nu; // contains now: a = dx.dot(H.dot(dx)) + rho * norm(dx)**2 +
98 // (mu_eq_inv) * norm(Adx)**2 + nu*mu_eq_inv * norm(Adx-dy*mu_eq)**2
99 qpwork.err.head(qpmodel.dim) =
100 qpresults.info.rho * (qpresults.x - qpwork.x_prev) + qpwork.g_scaled;
101 T b(qpresults.x.dot(qpwork.Hdx) +
102 (qpwork.err.head(qpmodel.dim)).dot(qpwork.dw_aug.head(qpmodel.dim)) +
103 qpresults.info.mu_eq_inv *
104 (qpwork.Adx)
105 .dot(qpwork.primal_residual_eq_scaled +
106 qpresults.y * qpresults.info.mu_eq)); // contains now: b =
107 // dx.dot(H.dot(x) +
108 // rho*(x-xe) + g) +
109 // mu_eq_inv * Adx.dot(res_eq)
110
111 qpwork.rhs.segment(qpmodel.dim, qpmodel.n_eq) =
112 qpwork.primal_residual_eq_scaled;
113 b += qpresults.info.nu * qpresults.info.mu_eq_inv *
114 qpwork.err.segment(qpmodel.dim, qpmodel.n_eq)
115 .dot(qpwork.rhs.segment(
116 qpmodel.dim,
117 qpmodel.n_eq)); // contains now: b = dx.dot(H.dot(x) + rho*(x-xe)
118 // + g) + mu_eq_inv * Adx.dot(res_eq) + nu*mu_eq_inv *
119 // (Adx-dy*mu_eq).dot(res_eq-y*mu_eq)
120
121 // derive Cdx_act
122 qpwork.err.tail(qpmodel.n_in) =
123 ((qpwork.primal_residual_in_scaled_up_plus_alphaCdx.array() > T(0.)) ||
124 (qpwork.primal_residual_in_scaled_low_plus_alphaCdx.array() < T(0.)))
125 .select(qpwork.Cdx,
126 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.n_in));
127
128 a += qpresults.info.mu_in_inv *
129 qpwork.err.tail(qpmodel.n_in)
130 .squaredNorm(); // contains now: a = dx.dot(H.dot(dx)) + rho *
131 // norm(dx)**2 + (mu_eq_inv) * norm(Adx)**2 + nu*mu_eq_inv *
132 // norm(Adx-dy*mu_eq)**2 + mu_in *
133 // norm(Cdx_act)**2
134
135 // derive vector [Cx-u+ze/mu]_+ + [Cx-l+ze/mu]--
136 qpwork.active_part_z =
137 (qpwork.primal_residual_in_scaled_up_plus_alphaCdx.array() > T(0.))
138 .select(qpwork.primal_residual_in_scaled_up,
139 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.n_in)) +
140 (qpwork.primal_residual_in_scaled_low_plus_alphaCdx.array() < T(0.))
141 .select(qpwork.primal_residual_in_scaled_low,
142 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(qpmodel.n_in));
143
144 b += qpresults.info.mu_in_inv *
145 qpwork.active_part_z.dot(qpwork.err.tail(
146 qpmodel.n_in)); // contains now: b = dx.dot(H.dot(x) + rho*(x-xe) +
147 // g) + mu_eq_inv * Adx.dot(res_eq) + nu*mu_eq_inv *
148 // (Adx-dy*mu_eq).dot(res_eq-y*mu_eq) + mu_in
149 // * Cdx_act.dot([Cx-u+ze/mu]_+ + [Cx-l+ze*mu_in]--)
150
151 // derive Cdx_act - dz*mu_in
152 qpwork.err.tail(qpmodel.n_in) -=
153 qpwork.dw_aug.tail(qpmodel.n_in) * qpresults.info.mu_in;
154 // derive [Cx-u+ze*mu_in]_+ + [Cx-l+ze*mu_in]-- -z*mu_in
155 qpwork.active_part_z -= qpresults.z * qpresults.info.mu_in;
156
157 // contains now a = dx.dot(H.dot(dx)) + rho * norm(dx)**2 + (mu_eq_inv) *
158 // norm(Adx)**2 + nu*mu_eq_inv * norm(Adx-dy*mu_eq)**2 + mu_in_inv *
159 // norm(Cdx_act)**2 + nu*mu_in_inv * norm(Cdx_act-dz*mu_in)**2
160 a += qpresults.info.nu * qpresults.info.mu_in_inv *
161 qpwork.err.tail(qpmodel.n_in).squaredNorm();
162 // contains now b = dx.dot(H.dot(x) + rho*(x-xe) + g) + mu_eq_inv *
163 // Adx.dot(res_eq) + nu*mu_eq_inv * (Adx-dy*mu_eq).dot(res_eq-y*mu_eq) +
164 // mu_in_inv
165 // * Cdx_act.dot([Cx-u+ze*mu_in]_+ + [Cx-l+ze*mu_in]--) + nu*mu_in_inv
166 // (Cdx_act-dz*mu_in).dot([Cx-u+ze*mu_in]_+ + [Cx-l+ze*mu_in]-- - z*mu_in)
167 b += qpresults.info.nu * qpresults.info.mu_in_inv *
168 qpwork.err.tail(qpmodel.n_in).dot(qpwork.active_part_z);
169
170 return {
171 a,
172 b,
173 a * alpha + b,
174 };
175}

185template<typename T>
186void
187primal_dual_ls(const Model<T>& qpmodel,
188 Results<T>& qpresults,
189 Workspace<T>& qpwork)
190{
191
192 /*
193 * The algorithm performs the following step
194 *
195 * 1/
196 * 1.1/ Store solutions of equations
197 * C(x+alpha dx) - l + ze/mu_in = 0
198 * C(x+alpha dx) - u + ze/mu_in = 0
199 *
200 * 1.2/ Sort the alpha
201 * 2/
202 * 2.1
203 * For each positive alpha compute the first derivative of
204 * phi(alpha) = [proximal primal dual augmented lagrangian of the subproblem
205 * evaluated at x_k + alpha dx, y_k + alpha dy, z_k + alpha dz] using function
206 * "gradient_norm" By construction for alpha = 0, phi'(alpha) <= 0 and
207 * phi'(alpha) goes to infinity with alpha hence it cancels uniquely at one
208 * optimal alpha*
209 *
210 * while phi'(alpha)<=0 store the derivative (noted last_grad_neg) and
211 * alpha (last_alpha_neg)
212 * the first time phi'(alpha) > 0 store the derivative (noted
213 * first_grad_pos) and alpha (first_alpha_pos), and break the loo
214 *
215 * 2.2
216 * If first_alpha_pos corresponds to the first positive alpha of previous
217 * loop, then do
218 * last_alpha_neg = 0
219 * last_grad_neg = phi'(0)
220 *
221 * 2.3
222 * the optimal alpha is within the interval
223 * [last_alpha_neg,first_alpha_pos] and can be computed exactly as phi' is
224 * an affine function in alph
225 * alpha* = alpha_last_neg
226 * - last_neg_grad * (alpha_first_pos - alpha_last_neg) /
227 * (first_pos_grad - last_neg_grad);
228 */
229
230 const T machine_eps = std::numeric_limits<T>::epsilon();
231
232 qpwork.alpha = T(1);
233 T alpha_(1.);
234
235 qpwork.alphas.clear();
236
238 // 1.1 add solutions of equations C(x+alpha dx)-l +ze/mu_in = 0 and C(x+alpha
239 // dx)-u +ze/mu_in = 0
240
241 for (isize i = 0; i < qpmodel.n_in; i++) {
242
243 if (qpwork.Cdx(i) != 0.) {
244 alpha_ =
245 -qpwork.primal_residual_in_scaled_up(i) / (qpwork.Cdx(i) + machine_eps);
246 if (alpha_ > machine_eps) {
247 qpwork.alphas.push(alpha_);
248 }
249 alpha_ = -qpwork.primal_residual_in_scaled_low(i) /
250 (qpwork.Cdx(i) + machine_eps);
251 if (alpha_ > machine_eps) {
252 qpwork.alphas.push(alpha_);
253 }
254 }
255 }
256
257 isize n_alpha = qpwork.alphas.len();
258
259 // 1.2 sort the alphas
260
261 std::sort(qpwork.alphas.ptr_mut(), qpwork.alphas.ptr_mut() + n_alpha);
262 isize new_len = std::unique( //
263 qpwork.alphas.ptr_mut(),
264 qpwork.alphas.ptr_mut() + n_alpha) -
265 qpwork.alphas.ptr_mut();
266 qpwork.alphas.resize(new_len);
267
268 n_alpha = qpwork.alphas.len();
269
270 if (n_alpha == 0 || qpwork.alphas[0] > 1) {
271 qpwork.alpha = 1;
272 return;
273 }
274
276 auto infty = std::numeric_limits<T>::infinity();
277
278 T last_neg_grad = 0;
279 T alpha_last_neg = 0;
280 T first_pos_grad = 0;
281 T alpha_first_pos = infty;
282 for (isize i = 0; i < n_alpha; ++i) {
283 alpha_ = qpwork.alphas[i];
284
285 /*
286 * 2.1
287 * For each positive alpha compute the first derivative of
288 * phi(alpha) = [proximal augmented lagrangian of the
289 * subproblem evaluated at x_k + alpha dx]
290 *
291 * (By construction for alpha = 0, phi'(alpha) <= 0 and
292 * phi'(alpha) goes to infinity with alpha hence it cancels
293 * uniquely at one optimal alpha*
294 *
295 * while phi'(alpha)<=0 store the derivative (noted
296 * last_grad_neg) and alpha (last_alpha_neg
297 * the first time phi'(alpha) > 0 store the derivative
298 * (noted first_grad_pos) and alpha (first_alpha_pos), and
299 * break the loop
300 */
301 T gr =
302 primal_dual_derivative_results(qpmodel, qpresults, qpwork, alpha_).grad;
303
304 if (gr < T(0)) {
305 alpha_last_neg = alpha_;
306 last_neg_grad = gr;
307 } else {
308 first_pos_grad = gr;
309 alpha_first_pos = alpha_;
310 break;
311 }
312 }
313
314 /*
315 * 2.2
316 * If first_alpha_pos corresponds to the first positive alpha of
317 * previous loop, then do
318 * last_alpha_neg = 0 and last_grad_neg = phi'(0) using function
319 * "gradient_norm"
320 */
321 if (alpha_last_neg == T(0)) {
322 last_neg_grad =
323 primal_dual_derivative_results(qpmodel, qpresults, qpwork, alpha_last_neg)
324 .grad;
325 }
326 if (alpha_first_pos == infty) {
327 /*
328 * 2.3
329 * the optimal alpha is within the interval
330 * [last_alpha_neg, +∞)
331 */
332 PrimalDualDerivativeResult<T> res = primal_dual_derivative_results(
333 qpmodel, qpresults, qpwork, 2 * alpha_last_neg + 1);
334 auto& a = res.a;
335 auto& b = res.b;
336 // grad = a * alpha + b
337 // grad = 0 => alpha = -b/a
338 qpwork.alpha = -b / a;
339 } else {
340 /*
341 * 2.3
342 * the optimal alpha is within the interval
343 * [last_alpha_neg,first_alpha_pos] and can be computed exactly as phi'
344 * is an affine function in alpha
345 */
346
347 qpwork.alpha = alpha_last_neg - last_neg_grad *
348 (alpha_first_pos - alpha_last_neg) /
349 (first_pos_grad - last_neg_grad);
350 }
351}

362template<typename T>
363void
364active_set_change(const Model<T>& qpmodel,
365 Results<T>& qpresults,
366 Workspace<T>& qpwork)
367{
368
369 /*
370 * arguments
371 * 1/ new_active_set : a vector which contains new active set of the
372 * problem, namely if
373 * new_active_set_u = Cx_k-u +z_k*mu_in>= 0
374 * new_active_set_l = Cx_k-l +z_k*mu_in<=
375 * then new_active_set = new_active_set_u OR new_active_set_
376 *
377 * 2/ current_bijection_map : a vector for which each entry corresponds to
378 * the current row of C of the current factorization
379 *
380 * for example, naming C_initial the initial C matrix of the problem, and
381 * C_current the one of the current factorization, the
382 * C_initial[i,:] = C_current[current_bijection_mal[i],:] for all
383 *
384 * 3/ n_c : the current number of active_inequalities
385 * This algorithm ensures that for all new version of C_current in the LDLT
386 * factorization all row index i < n_c correspond to current active indexes
387 * (all other correspond to inactive rows
388 *
389 * To do so,
390 * 1/ for initialization
391 * 1.1/ new_bijection_map = current_bijection_map
392 * 1.2/ n_c_f = n_
393 *
394 * 2/ All active indexes of the current bijection map (i.e
395 * current_bijection_map(i) < n_c by assumption) which are not active
396 * anymore in the new active set (new_active_set(i)=false are put at the
397 * end of new_bijection_map, i.
398 *
399 * 2.1/ for all j if new_bijection_map(j) > new_bijection_map(i), then
400 * new_bijection_map(j)-=1
401 * 2.2/ n_c_f -=1
402 * 2.3/ new_bijection_map(i) = n_in-1
403 *
404 * 3/ All active indexe of the new active set (new_active_set(i) == true)
405 * which are not active in the new_bijection_map (new_bijection_map(i) >=
406 * n_c_f) are put at the end of the current version of C, i.e
407 * 3.1/ if new_bijection_map(j) < new_bijection_map(i) &&
408 * new_bijection_map(j) >= n_c_f then new_bijection_map(j)+=1
409 * 3.2/ new_bijection_map(i) = n_c_f
410 * 3.3/ n_c_f +=1
411 *
412 * It returns finally the new_bijection_map, for which
413 * new_bijection_map(n_in) = n_c_f
414 */
415
416 qpwork.dw_aug.setZero();
417
418 isize n_c_f = qpwork.n_c;
419 qpwork.new_bijection_map = qpwork.current_bijection_map;
420
421 // suppression pour le nouvel active set, ajout dans le nouvel unactive set
422
423 proxsuite::linalg::veg::dynstack::DynStackMut stack{
424 proxsuite::linalg::veg::from_slice_mut, qpwork.ldl_stack.as_mut()
425 };
426
427 {
428 auto _planned_to_delete = stack.make_new_for_overwrite(
429 proxsuite::linalg::veg::Tag<isize>{}, isize(qpmodel.n_in));
430 isize* planned_to_delete = _planned_to_delete.ptr_mut();
431 isize planned_to_delete_count = 0;
432
433 for (isize i = 0; i < qpmodel.n_in; i++) {
434 if (qpwork.current_bijection_map(i) < qpwork.n_c) {
435 if (!qpwork.active_inequalities(i)) {
436 // delete current_bijection_map(i)
437
438 planned_to_delete[planned_to_delete_count] =
439 qpwork.current_bijection_map(i) + qpmodel.dim + qpmodel.n_eq;
440 ++planned_to_delete_count;
441
442 for (isize j = 0; j < qpmodel.n_in; j++) {
443 if (qpwork.new_bijection_map(j) > qpwork.new_bijection_map(i)) {
444 qpwork.new_bijection_map(j) -= 1;
445 }
446 }
447 n_c_f -= 1;
448 qpwork.new_bijection_map(i) = qpmodel.n_in - 1;
449 }
450 }
451 }
452 std::sort(planned_to_delete, planned_to_delete + planned_to_delete_count);
453 qpwork.ldl.delete_at(planned_to_delete, planned_to_delete_count, stack);
454 if (planned_to_delete_count > 0) {
455 qpwork.constraints_changed = true;
456 }
457 }
458
459 // ajout au nouvel active set, suppression pour le nouvel unactive set
460
461 {
462 auto _planned_to_add = stack.make_new_for_overwrite(
463 proxsuite::linalg::veg::Tag<isize>{}, qpmodel.n_in);
464 auto planned_to_add = _planned_to_add.ptr_mut();
465
466 isize planned_to_add_count = 0;
467 T mu_in_neg = -qpresults.info.mu_in;
468 isize n_c = n_c_f;
469 for (isize i = 0; i < qpmodel.n_in; i++) {
470 if (qpwork.active_inequalities(i)) {
471 if (qpwork.new_bijection_map(i) >= n_c_f) {
472 // add at the end
473 planned_to_add[planned_to_add_count] = i;
474 ++planned_to_add_count;
475
476 for (isize j = 0; j < qpmodel.n_in; j++) {
477 if (qpwork.new_bijection_map(j) < qpwork.new_bijection_map(i) &&
478 qpwork.new_bijection_map(j) >= n_c_f) {
479 qpwork.new_bijection_map(j) += 1;
480 }
481 }
482 qpwork.new_bijection_map(i) = n_c_f;
483 n_c_f += 1;
484 }
485 }
486 }
487 {
488 isize n = qpmodel.dim;
489 isize n_eq = qpmodel.n_eq;
490 LDLT_TEMP_MAT_UNINIT(
491 T, new_cols, n + n_eq + n_c_f, planned_to_add_count, stack);
492
493 for (isize k = 0; k < planned_to_add_count; ++k) {
494 isize index = planned_to_add[k];
495 auto col = new_cols.col(k);
496 col.head(n) = (qpwork.C_scaled.row(index));
497 col.tail(n_eq + n_c_f).setZero();
498 col[n + n_eq + n_c + k] = mu_in_neg;
499 }
500 qpwork.ldl.insert_block_at(n + n_eq + n_c, new_cols, stack);
501 }
502 if (planned_to_add_count > 0) {
503 qpwork.constraints_changed = true;
504 }
505 }
506
507 qpwork.n_c = n_c_f;
508 qpwork.current_bijection_map = qpwork.new_bijection_map;
509 qpwork.dw_aug.setZero();
510}
