30template<typename T>
31struct PrimalDualDerivativeResult
32{
33 T a;
34 T b;
35 T grad;
36 VEG_REFLECT(PrimalDualDerivativeResult, a, b, grad);
37};
49template<typename T>
50auto
51gpdal_derivative_results(const Model<T>& qpmodel,
52 Results<T>& qpresults,
53 Workspace<T>& qpwork,
54 const Settings<T>& qpsettings,
55 isize n_constraints,
56 T alpha) -> PrimalDualDerivativeResult<T>
57{
58
59 /*
60 * the function computes the first derivative of phi(alpha) at outer step k
61 * and inner step l
62 *
63 * phi(alpha) = f(x_l+alpha dx) + rho/2 |x_l + alpha dx - x_k|**2
64 * + mu_eq_inv/2 (|A(x_l+alpha dx)-d+y_k * mu_eq|**2)
65 * + mu_eq_inv * nu /2 (|A(x_l+alpha dx)-d+y_k * mu_eq -
66 * (y_l+alpha dy)
67 * |**2)
68 * + mu_in_inv/2 ( | [C(x_l+alpha dx) - u + z_k * mu_in]_+ |**2
69 * +| [C(x_l+alpha dx) - l + z_k * mu_in]_- |**2
70 * )
71 * + mu_in_inv * nu / 2 ( | [C(x_l+alpha dx) - u +
72 * z_k
73 * * mu_in]_+
74 * + [C(x_l+alpha dx) - l + z_k * mu_in]_- - (z+alpha dz) * mu_in |**2 with
75 * f(x) = 0.5 * x^THx + g^Tx phi is a second order polynomial in alpha. Below
76 * are computed its coefficients a0 and b0 in order to compute the desired
77 * gradient a0 * alpha + b0
78 */
79
80 qpwork.primal_residual_in_scaled_up_plus_alphaCdx =
81 qpwork.primal_residual_in_scaled_up + qpwork.Cdx * alpha;
82 qpwork.primal_residual_in_scaled_low_plus_alphaCdx =
83 qpresults.si + qpwork.Cdx * alpha;
84
85 T a(qpwork.dw_aug.head(qpmodel.dim).dot(qpwork.Hdx) +
86 qpresults.info.mu_eq_inv * (qpwork.Adx).squaredNorm() +
87 qpresults.info.rho *
88 qpwork.dw_aug.head(qpmodel.dim)
89 .squaredNorm()); // contains now: a = dx.dot(H.dot(dx)) + rho *
90 // norm(dx)**2 + (mu_eq_inv) * norm(Adx)**2
91
92 qpwork.err.segment(qpmodel.dim, qpmodel.n_eq) =
93 qpwork.Adx -
94 qpwork.dw_aug.segment(qpmodel.dim, qpmodel.n_eq) * qpresults.info.mu_eq;
95 a +=
96 qpwork.err.segment(qpmodel.dim, qpmodel.n_eq).squaredNorm() *
97 qpresults.info
98 .mu_eq_inv; // contains now: a = dx.dot(H.dot(dx)) + rho * norm(dx)**2 +
99 // (mu_eq_inv) * norm(Adx)**2 + mu_eq_inv * norm(Adx-dy*mu_eq)**2
100 qpwork.err.head(qpmodel.dim) =
101 qpresults.info.rho * (qpresults.x - qpwork.x_prev) + qpwork.g_scaled;
102 T b(qpresults.x.dot(qpwork.Hdx) +
103 (qpwork.err.head(qpmodel.dim)).dot(qpwork.dw_aug.head(qpmodel.dim)) +
104 qpresults.info.mu_eq_inv *
105 (qpwork.Adx)
106 .dot(qpresults.se +
107 qpresults.y * qpresults.info.mu_eq)); // contains now: b =
108 // dx.dot(H.dot(x) +
109 // rho*(x-xe) + g) +
110 // mu_eq_inv * Adx.dot(res_eq)
111
112 qpwork.rhs.segment(qpmodel.dim, qpmodel.n_eq) = qpresults.se;
113 b += qpresults.info.mu_eq_inv *
114 qpwork.err.segment(qpmodel.dim, qpmodel.n_eq)
115 .dot(qpwork.rhs.segment(
116 qpmodel.dim,
117 qpmodel.n_eq)); // contains now: b = dx.dot(H.dot(x) + rho*(x-xe)
118 // + g) + mu_eq_inv * Adx.dot(res_eq) + nu*mu_eq_inv *
119 // (Adx-dy*mu_eq).dot(res_eq-y*mu_eq)
120
121 // derive Cdx_act
122 qpwork.err.tail(n_constraints) =
123 ((qpwork.primal_residual_in_scaled_up_plus_alphaCdx.array() > T(0.)) ||
124 (qpwork.primal_residual_in_scaled_low_plus_alphaCdx.array() < T(0.)))
125 .select(qpwork.Cdx,
126 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(n_constraints));
127
128 a += qpresults.info.mu_in_inv * qpwork.err.tail(n_constraints).squaredNorm() /
129 qpsettings.alpha_gpdal; // contains now: a = dx.dot(H.dot(dx)) + rho *
130 // norm(dx)**2 + (mu_eq_inv) * norm(Adx)**2 + nu*mu_eq_inv *
131 // norm(Adx-dy*mu_eq)**2 +
132 // norm(dw_act)**2 / (mu_in * (alpha_gpdal))
133 a += qpresults.info.mu_in * (1. - qpsettings.alpha_gpdal) *
134 qpwork.dw_aug.tail(n_constraints).squaredNorm();
135 // add norm(z)**2 * mu_in * (1-alpha)
136
137 // derive vector [w-u]_+ + [w-l]--
138 qpwork.active_part_z =
139 (qpwork.primal_residual_in_scaled_up_plus_alphaCdx.array() > T(0.))
140 .select(qpwork.primal_residual_in_scaled_up,
141 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(n_constraints)) +
142 (qpwork.primal_residual_in_scaled_low_plus_alphaCdx.array() < T(0.))
143 .select(qpresults.si,
144 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(n_constraints));
145
146 b +=
147 qpresults.info.mu_in_inv *
148 qpwork.active_part_z.dot(qpwork.err.tail(n_constraints)) /
149 qpsettings.alpha_gpdal; // contains now: b = dx.dot(H.dot(x) + rho*(x-xe) +
150 // g) + mu_eq_inv * Adx.dot(res_eq) + nu*mu_eq_inv *
151 // (Adx-dy*mu_eq).dot(res_eq-y*mu_eq) + mu_in
152 // * dw_act.dot([w-u]_+ + [w-l]--) / alpha_gpdal
153
154 // contains now b = dx.dot(H.dot(x) + rho*(x-xe) + g) + mu_eq_inv *
155 // Adx.dot(res_eq) + nu*mu_eq_inv * (Adx-dy*mu_eq).dot(res_eq-y*mu_eq) +
156 // mu_in_inv
157 // * Cdx_act.dot([Cx-u+ze*mu_in]_+ + [Cx-l+ze*mu_in]--) + nu*mu_in_inv
158 // (Cdx_act-dz*mu_in).dot([Cx-u+ze*mu_in]_+ + [Cx-l+ze*mu_in]-- - z*mu_in)
159 b += qpresults.info.mu_in * (1. - qpsettings.alpha_gpdal) *
160 qpwork.dw_aug.tail(n_constraints).dot(qpresults.z);
161
162 return {
163 a,
164 b,
165 a * alpha + b,
166 };
167}
178template<typename T>
179auto
180primal_dual_derivative_results(const Model<T>& qpmodel,
181 Results<T>& qpresults,
182 Workspace<T>& qpwork,
183 isize n_constraints,
184 T alpha) -> PrimalDualDerivativeResult<T>
185{
186
187 /*
188 * the function computes the first derivative of phi(alpha) at outer step k
189 * and inner step l
190 *
191 * phi(alpha) = f(x_l+alpha dx) + rho/2 |x_l + alpha dx - x_k|**2
192 * + mu_eq_inv/2 (|A(x_l+alpha dx)-d+y_k * mu_eq|**2)
193 * + mu_eq_inv * nu /2 (|A(x_l+alpha dx)-d+y_k * mu_eq -
194 * (y_l+alpha dy)
195 * |**2)
196 * + mu_in_inv/2 ( | [C(x_l+alpha dx) - u + z_k * mu_in]_+ |**2
197 * +| [C(x_l+alpha dx) - l + z_k * mu_in]_- |**2
198 * )
199 * + mu_in_inv * nu / 2 ( | [C(x_l+alpha dx) - u +
200 * z_k
201 * * mu_in]_+
202 * + [C(x_l+alpha dx) - l + z_k * mu_in]_- - (z+alpha dz) * mu_in |**2 with
203 * f(x) = 0.5 * x^THx + g^Tx phi is a second order polynomial in alpha. Below
204 * are computed its coefficients a0 and b0 in order to compute the desired
205 * gradient a0 * alpha + b0
206 */
207
208 qpwork.primal_residual_in_scaled_up_plus_alphaCdx =
209 qpwork.primal_residual_in_scaled_up + qpwork.Cdx * alpha;
210 qpwork.primal_residual_in_scaled_low_plus_alphaCdx =
211 qpresults.si + qpwork.Cdx * alpha;
212
213 T a(qpwork.dw_aug.head(qpmodel.dim).dot(qpwork.Hdx) +
214 qpresults.info.mu_eq_inv * (qpwork.Adx).squaredNorm() +
215 qpresults.info.rho *
216 qpwork.dw_aug.head(qpmodel.dim)
217 .squaredNorm()); // contains now: a = dx.dot(H.dot(dx)) + rho *
218 // norm(dx)**2 + (mu_eq_inv) * norm(Adx)**2
219
220 qpwork.err.segment(qpmodel.dim, qpmodel.n_eq) =
221 qpwork.Adx -
222 qpwork.dw_aug.segment(qpmodel.dim, qpmodel.n_eq) * qpresults.info.mu_eq;
223 a += qpwork.err.segment(qpmodel.dim, qpmodel.n_eq).squaredNorm() *
224 qpresults.info.mu_eq_inv *
225 qpresults.info
226 .nu; // contains now: a = dx.dot(H.dot(dx)) + rho * norm(dx)**2 +
227 // (mu_eq_inv) * norm(Adx)**2 + nu*mu_eq_inv * norm(Adx-dy*mu_eq)**2
228 qpwork.err.head(qpmodel.dim) =
229 qpresults.info.rho * (qpresults.x - qpwork.x_prev) + qpwork.g_scaled;
230 T b(qpresults.x.dot(qpwork.Hdx) +
231 (qpwork.err.head(qpmodel.dim)).dot(qpwork.dw_aug.head(qpmodel.dim)) +
232 qpresults.info.mu_eq_inv *
233 (qpwork.Adx)
234 .dot(qpresults.se +
235 qpresults.y * qpresults.info.mu_eq)); // contains now: b =
236 // dx.dot(H.dot(x) +
237 // rho*(x-xe) + g) +
238 // mu_eq_inv * Adx.dot(res_eq)
239
240 qpwork.rhs.segment(qpmodel.dim, qpmodel.n_eq) = qpresults.se;
241 b += qpresults.info.nu * qpresults.info.mu_eq_inv *
242 qpwork.err.segment(qpmodel.dim, qpmodel.n_eq)
243 .dot(qpwork.rhs.segment(
244 qpmodel.dim,
245 qpmodel.n_eq)); // contains now: b = dx.dot(H.dot(x) + rho*(x-xe)
246 // + g) + mu_eq_inv * Adx.dot(res_eq) + nu*mu_eq_inv *
247 // (Adx-dy*mu_eq).dot(res_eq-y*mu_eq)
248
249 // derive Cdx_act
250 qpwork.err.tail(n_constraints) =
251 ((qpwork.primal_residual_in_scaled_up_plus_alphaCdx.array() > T(0.)) ||
252 (qpwork.primal_residual_in_scaled_low_plus_alphaCdx.array() < T(0.)))
253 .select(qpwork.Cdx,
254 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(n_constraints));
255
256 a += qpresults.info.mu_in_inv *
257 qpwork.err.tail(n_constraints)
258 .squaredNorm(); // contains now: a = dx.dot(H.dot(dx)) + rho *
259 // norm(dx)**2 + (mu_eq_inv) * norm(Adx)**2 + nu*mu_eq_inv *
260 // norm(Adx-dy*mu_eq)**2 + mu_in *
261 // norm(Cdx_act)**2
262
263 // derive vector [Cx-u+ze/mu]_+ + [Cx-l+ze/mu]--
264 qpwork.active_part_z =
265 (qpwork.primal_residual_in_scaled_up_plus_alphaCdx.array() > T(0.))
266 .select(qpwork.primal_residual_in_scaled_up,
267 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(n_constraints)) +
268 (qpwork.primal_residual_in_scaled_low_plus_alphaCdx.array() < T(0.))
269 .select(qpresults.si,
270 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(n_constraints));
271
272 b += qpresults.info.mu_in_inv *
273 qpwork.active_part_z.dot(qpwork.err.tail(
274 n_constraints)); // contains now: b = dx.dot(H.dot(x) + rho*(x-xe) +
275 // g) + mu_eq_inv * Adx.dot(res_eq) + nu*mu_eq_inv *
276 // (Adx-dy*mu_eq).dot(res_eq-y*mu_eq) + mu_in
277 // * Cdx_act.dot([Cx-u+ze/mu]_+ + [Cx-l+ze*mu_in]--)
278
279 // derive Cdx_act - dz*mu_in
280 qpwork.err.tail(n_constraints) -=
281 qpwork.dw_aug.tail(n_constraints) * qpresults.info.mu_in;
282 // derive [Cx-u+ze*mu_in]_+ + [Cx-l+ze*mu_in]-- -z*mu_in
283 qpwork.active_part_z -= qpresults.z * qpresults.info.mu_in;
284
285 // contains now a = dx.dot(H.dot(dx)) + rho * norm(dx)**2 + (mu_eq_inv) *
286 // norm(Adx)**2 + nu*mu_eq_inv * norm(Adx-dy*mu_eq)**2 + mu_in_inv *
287 // norm(Cdx_act)**2 + nu*mu_in_inv * norm(Cdx_act-dz*mu_in)**2
288 a += qpresults.info.nu * qpresults.info.mu_in_inv *
289 qpwork.err.tail(n_constraints).squaredNorm();
290 // contains now b = dx.dot(H.dot(x) + rho*(x-xe) + g) + mu_eq_inv *
291 // Adx.dot(res_eq) + nu*mu_eq_inv * (Adx-dy*mu_eq).dot(res_eq-y*mu_eq) +
292 // mu_in_inv
293 // * Cdx_act.dot([Cx-u+ze*mu_in]_+ + [Cx-l+ze*mu_in]--) + nu*mu_in_inv
294 // (Cdx_act-dz*mu_in).dot([Cx-u+ze*mu_in]_+ + [Cx-l+ze*mu_in]-- - z*mu_in)
295 b += qpresults.info.nu * qpresults.info.mu_in_inv *
296 qpwork.err.tail(n_constraints).dot(qpwork.active_part_z);
297
298 return {
299 a,
300 b,
301 a * alpha + b,
302 };

312template<typename T>
313void
314primal_dual_ls(const Model<T>& qpmodel,
315 Results<T>& qpresults,
316 Workspace<T>& qpwork,
317 const Settings<T>& qpsettings,
318 const isize n_constraints)
319{
320
321 /*
322 * The algorithm performs the following step
323 *
324 * 1/
325 * 1.1/ Store solutions of equations
326 * C(x+alpha dx) - l + ze/mu_in = 0
327 * C(x+alpha dx) - u + ze/mu_in = 0
328 *
329 * 1.2/ Sort the alpha
330 * 2/
331 * 2.1
332 * For each positive alpha compute the first derivative of
333 * phi(alpha) = [proximal primal dual augmented lagrangian of the subproblem
334 * evaluated at x_k + alpha dx, y_k + alpha dy, z_k + alpha dz] using function
335 * "gradient_norm" By construction for alpha = 0, phi'(alpha) <= 0 and
336 * phi'(alpha) goes to infinity with alpha hence it cancels uniquely at one
337 * optimal alpha*
338 *
339 * while phi'(alpha)<=0 store the derivative (noted last_grad_neg) and
340 * alpha (last_alpha_neg)
341 * the first time phi'(alpha) > 0 store the derivative (noted
342 * first_grad_pos) and alpha (first_alpha_pos), and break the loo
343 *
344 * 2.2
345 * If first_alpha_pos corresponds to the first positive alpha of previous
346 * loop, then do
347 * last_alpha_neg = 0
348 * last_grad_neg = phi'(0)
349 *
350 * 2.3
351 * the optimal alpha is within the interval
352 * [last_alpha_neg,first_alpha_pos] and can be computed exactly as phi' is
353 * an affine function in alph
354 * alpha* = alpha_last_neg
355 * - last_neg_grad * (alpha_first_pos - alpha_last_neg) /
356 * (first_pos_grad - last_neg_grad);
357 */
358
359 const T machine_eps = std::numeric_limits<T>::epsilon();
360
361 qpwork.alpha = T(1);
362 T alpha_(1.);
363
364 qpwork.alphas.clear();
365
367 // 1.1 add solutions of equations C(x+alpha dx)-l +ze/mu_in = 0 and C(x+alpha
368 // dx)-u +ze/mu_in = 0
369
370 for (isize i = 0; i < n_constraints; i++) {
371
372 if (qpwork.Cdx(i) != 0.) {
373 alpha_ =
374 -qpwork.primal_residual_in_scaled_up(i) / (qpwork.Cdx(i) + machine_eps);
375 if (alpha_ > machine_eps) {
376 qpwork.alphas.push(alpha_);
377 }
378 alpha_ = -qpresults.si(i) / (qpwork.Cdx(i) + machine_eps);
379 if (alpha_ > machine_eps) {
380 qpwork.alphas.push(alpha_);
381 }
382 }
383 }
384
385 isize n_alpha = qpwork.alphas.len();
386
387 // 1.2 sort the alphas
388
389 std::sort(qpwork.alphas.ptr_mut(), qpwork.alphas.ptr_mut() + n_alpha);
390 isize new_len = std::unique( //
391 qpwork.alphas.ptr_mut(),
392 qpwork.alphas.ptr_mut() + n_alpha) -
393 qpwork.alphas.ptr_mut();
394 qpwork.alphas.resize(new_len);
395
396 n_alpha = qpwork.alphas.len();
397 if (n_alpha == 0) { //
398 switch (qpsettings.merit_function_type) {
399 case MeritFunctionType::GPDAL: {
400 auto res = gpdal_derivative_results(
401 qpmodel, qpresults, qpwork, qpsettings, n_constraints, T(0));
402 qpwork.alpha = -res.b / res.a;
403 } break;
404 case MeritFunctionType::PDAL: {
405 auto res = primal_dual_derivative_results(
406 qpmodel, qpresults, qpwork, n_constraints, T(0));
407 qpwork.alpha = -res.b / res.a;
408 } break;
409 }
410 return;
411 }
413 auto infty = std::numeric_limits<T>::infinity();
414
415 T last_neg_grad = 0;
416 T alpha_last_neg = 0;
417 T first_pos_grad = 0;
418 T alpha_first_pos = infty;
419 for (isize i = 0; i < n_alpha; ++i) {
420 alpha_ = qpwork.alphas[i];
421
422 /*
423 * 2.1
424 * For each positive alpha compute the first derivative of
425 * phi(alpha) = [proximal augmented lagrangian of the
426 * subproblem evaluated at x_k + alpha dx]
427 *
428 * (By construction for alpha = 0, phi'(alpha) <= 0 and
429 * phi'(alpha) goes to infinity with alpha hence it cancels
430 * uniquely at one optimal alpha*
431 *
432 * while phi'(alpha)<=0 store the derivative (noted
433 * last_grad_neg) and alpha (last_alpha_neg
434 * the first time phi'(alpha) > 0 store the derivative
435 * (noted first_grad_pos) and alpha (first_alpha_pos), and
436 * break the loop
437 */
438 T gr(0);
439 switch (qpsettings.merit_function_type) {
440 case MeritFunctionType::GPDAL:
441 gr = gpdal_derivative_results(
442 qpmodel, qpresults, qpwork, qpsettings, n_constraints, alpha_)
443 .grad;
444 break;
445 case MeritFunctionType::PDAL:
446 gr = primal_dual_derivative_results(
447 qpmodel, qpresults, qpwork, n_constraints, alpha_)
448 .grad;
449 break;
450 }
451
452 if (gr < T(0)) {
453 alpha_last_neg = alpha_;
454 last_neg_grad = gr;
455 } else {
456 first_pos_grad = gr;
457 alpha_first_pos = alpha_;
458 break;
459 }
460 }
461
462 /*
463 * 2.2
464 * If first_alpha_pos corresponds to the first positive alpha of
465 * previous loop, then do
466 * last_alpha_neg = 0 and last_grad_neg = phi'(0) using function
467 * "gradient_norm"
468 */
469 if (alpha_last_neg == T(0)) {
470 switch (qpsettings.merit_function_type) {
471 case MeritFunctionType::GPDAL:
472 last_neg_grad = gpdal_derivative_results(qpmodel,
473 qpresults,
474 qpwork,
475 qpsettings,
476 n_constraints,
477 alpha_last_neg)
478 .grad;
479 break;
480 case MeritFunctionType::PDAL:
481 last_neg_grad =
482 primal_dual_derivative_results(
483 qpmodel, qpresults, qpwork, n_constraints, alpha_last_neg)
484 .grad;
485 break;
486 }
487 }
488 if (alpha_first_pos == infty) {
489 /*
490 * 2.3
491 * the optimal alpha is within the interval
492 * [last_alpha_neg, +)
493 */
494 switch (qpsettings.merit_function_type) {
495 case MeritFunctionType::GPDAL: {
496 PrimalDualDerivativeResult<T> res =
497 gpdal_derivative_results(qpmodel,
498 qpresults,
499 qpwork,
500 qpsettings,
501 n_constraints,
502 2 * alpha_last_neg + 1);
503 auto& a = res.a;
504 auto& b = res.b;
505 // grad = a * alpha + b
506 // grad = 0 => alpha = -b/a
507 qpwork.alpha = -b / a;
508 } break;
509 case MeritFunctionType::PDAL: {
510 PrimalDualDerivativeResult<T> res = primal_dual_derivative_results(
511 qpmodel, qpresults, qpwork, n_constraints, 2 * alpha_last_neg + 1);
512 auto& a = res.a;
513 auto& b = res.b;
514 // grad = a * alpha + b
515 // grad = 0 => alpha = -b/a
516 qpwork.alpha = -b / a;
517 } break;
518 }
519 } else {
520 /*
521 * 2.3
522 * the optimal alpha is within the interval
523 * [last_alpha_neg,first_alpha_pos] and can be computed exactly as phi'
524 * is an affine function in alpha
525 */
526 qpwork.alpha = std::abs(alpha_last_neg -
527 last_neg_grad * (alpha_first_pos - alpha_last_neg) /
528 (first_pos_grad - last_neg_grad));
529 }
530}

541template<typename T>
542void
543active_set_change(const Model<T>& qpmodel,
544 Results<T>& qpresults,
545 const DenseBackend& dense_backend,
546 const isize n_constraints,
547 Workspace<T>& qpwork)
548{
549
550 /*
551 * arguments
552 * 1/ new_active_set : a vector which contains new active set of the
553 * problem, namely if
554 * new_active_set_u = Cx_k-u +z_k*mu_in>= 0
555 * new_active_set_l = Cx_k-l +z_k*mu_in<=
556 * then new_active_set = new_active_set_u OR new_active_set_
557 *
558 * 2/ current_bijection_map : a vector for which each entry corresponds to
559 * the current row of C of the current factorization
560 *
561 * for example, naming C_initial the initial C matrix of the problem, and
562 * C_current the one of the current factorization, the
563 * C_initial[i,:] = C_current[current_bijection_mal[i],:] for all
564 *
565 * 3/ n_c : the current number of active_inequalities
566 * This algorithm ensures that for all new version of C_current in the LDLT
567 * factorization all row index i < n_c correspond to current active indexes
568 * (all other correspond to inactive rows
569 *
570 * To do so,
571 * 1/ for initialization
572 * 1.1/ new_bijection_map = current_bijection_map
573 * 1.2/ n_c_f = n_
574 *
575 * 2/ All active indexes of the current bijection map (i.e
