111template<typename T, typename I>
112VEG_NO_INLINE void
113noalias_gevmmv_add_impl( //
114 VectorViewMut<T> out_l,
115 VectorViewMut<T> out_r,
116 proxsuite::linalg::sparse::MatRef<T, I> a,
117 VectorView<T> in_l,
118 VectorView<T> in_r)
119{
120 VEG_ASSERT_ALL_OF /* NOLINT */ (a.nrows() == out_r.dim,
121 a.ncols() == in_r.dim,
122 a.ncols() == out_l.dim,
123 a.nrows() == in_l.dim);
124 // equivalent to
125 // out_r.to_eigen().noalias() += a.to_eigen() * in_r.to_eigen();
126 // out_l.to_eigen().noalias() += a.to_eigen().transpose() * in_l.to_eigen();
127
128 auto* ai = a.row_indices();
129 auto* ax = a.values();
130 auto n = a.ncols();
131
132 for (usize j = 0; j < usize(n); ++j) {
133 usize col_start = a.col_start(j);
134 usize col_end = a.col_end(j);
135
136 T acc0 = 0;
137 T acc1 = 0;
138 T acc2 = 0;
139 T acc3 = 0;
140
141 T in_rj = in_r(isize(j));
142
143 usize pcount = col_end - col_start;
144
145 usize p = col_start;
146
147 auto zx = proxsuite::linalg::sparse::util::zero_extend;
148
149 for (; p < col_start + pcount / 4 * 4; p += 4) {
150 auto i0 = isize(zx(ai[p + 0]));
151 auto i1 = isize(zx(ai[p + 1]));
152 auto i2 = isize(zx(ai[p + 2]));
153 auto i3 = isize(zx(ai[p + 3]));
154
155 T ai0j = ax[p + 0];
156 T ai1j = ax[p + 1];
157 T ai2j = ax[p + 2];
158 T ai3j = ax[p + 3];
159
160 out_r(i0) += ai0j * in_rj;
161 out_r(i1) += ai1j * in_rj;
162 out_r(i2) += ai2j * in_rj;
163 out_r(i3) += ai3j * in_rj;
164
165 acc0 += ai0j * in_l(i0);
166 acc1 += ai1j * in_l(i1);
167 acc2 += ai2j * in_l(i2);
168 acc3 += ai3j * in_l(i3);
169 }
170
171 for (; p < col_end; ++p) {
172 auto i = isize(zx(ai[p]));
173
174 T aij = ax[p];
175 out_r(i) += aij * in_rj;
176 acc0 += aij * in_l(i);
177 }
178
179 acc0 = ((acc0 + acc1) + (acc2 + acc3));
180 out_l(isize(j)) += acc0;
181 }
182}

184template<typename T, typename I>
185VEG_NO_INLINE void
186noalias_symhiv_add_impl( //
187 VectorViewMut<T> out,
188 proxsuite::linalg::sparse::MatRef<T, I> a,
189 VectorView<T> in)
190{
191 VEG_ASSERT_ALL_OF /* NOLINT */ ( //
192 a.nrows() == a.ncols(),
193 a.nrows() == out.dim,
194 a.ncols() == in.dim);
195 // equivalent to
196 // out.to_eigen().noalias() +=
197 // a.to_eigen().template selfadjointView<Eigen::Upper>() *
198 // in.to_eigen();
199
200 auto* ai = a.row_indices();
201 auto* ax = a.values();
202 auto n = a.ncols();
203
204 for (usize j = 0; j < usize(n); ++j) {
205 usize col_start = a.col_start(j);
206 usize col_end = a.col_end(j);
207
208 if (col_start == col_end) {
209 continue;
210 }
211
212 T acc0 = 0;
213 T acc1 = 0;
214 T acc2 = 0;
215 T acc3 = 0;
216
217 T in_j = in(isize(j));
218
219 usize pcount = col_end - col_start;
220
221 auto zx = proxsuite::linalg::sparse::util::zero_extend;
222
223 if (zx(ai[col_end - 1]) == j) {
224 T ajj = ax[col_end - 1];
225 out(isize(j)) += ajj * in_j;
226 pcount -= 1;
227 }
228
229 usize p = col_start;
230
231 for (; p < col_start + pcount / 4 * 4; p += 4) {
232 auto i0 = isize(zx(ai[p + 0]));
233 auto i1 = isize(zx(ai[p + 1]));
234 auto i2 = isize(zx(ai[p + 2]));
235 auto i3 = isize(zx(ai[p + 3]));
236
237 T ai0j = ax[p + 0];
238 T ai1j = ax[p + 1];
239 T ai2j = ax[p + 2];
240 T ai3j = ax[p + 3];
241
242 out(i0) += ai0j * in_j;
243 out(i1) += ai1j * in_j;
244 out(i2) += ai2j * in_j;
245 out(i3) += ai3j * in_j;
246
247 acc0 += ai0j * in(i0);
248 acc1 += ai1j * in(i1);
249 acc2 += ai2j * in(i2);
250 acc3 += ai3j * in(i3);
251 }
252 for (; p < col_start + pcount; ++p) {
253 auto i = isize(zx(ai[p]));
254
255 T aij = ax[p];
256 out(i) += aij * in_j;
257 acc0 += aij * in(i);
258 }
259 acc0 = ((acc0 + acc1) + (acc2 + acc3));
260 out(isize(j)) += acc0;
261 }
262}

476template<typename T, typename I, typename P>
477bool
478global_primal_residual_infeasibility(VectorViewMut<T> ATdy,
479 VectorViewMut<T> CTdz,
480 VectorViewMut<T> dy,
481 VectorViewMut<T> dz,
482 QpView<T, I> qp_scaled,
483 const Settings<T>& qpsettings,
484 P& ruiz)
485{
486
487 // The problem is primal infeasible if the following four conditions hold:
488 //
489 // ||unscaled(A^Tdy)|| <= eps_p_inf ||unscaled(dy)||
490 // b^T dy <= -eps_p_inf ||unscaled(dy)||
491 // ||unscaled(C^Tdz)|| <= eps_p_inf ||unscaled(dz)||
492 // u^T [dz]_+ - l^T[-dz]_+ <= -eps_p_inf ||unscaled(dz)||
493 //
494 // the variables in entry are changed in place
495
496 bool res = infty_norm(dy.to_eigen()) != 0 && infty_norm(dz.to_eigen()) != 0;
497 if (!res) {
498 return res;
499 }
500 ruiz.unscale_dual_residual_in_place(ATdy);
501 ruiz.unscale_dual_residual_in_place(CTdz);
502 T eq_inf = dy.to_eigen().dot(qp_scaled.b.to_eigen());
503 T in_inf = positive_part(dz.to_eigen()).dot(qp_scaled.u.to_eigen()) -
504 positive_part(-dz.to_eigen()).dot(qp_scaled.l.to_eigen());
505 ruiz.unscale_dual_in_place_eq(dy);
506 ruiz.unscale_dual_in_place_in(dz);
507
508 T bound_y = qpsettings.eps_primal_inf * infty_norm(dy.to_eigen());
509 T bound_z = qpsettings.eps_primal_inf * infty_norm(dz.to_eigen());
510
511 res = infty_norm(ATdy.to_eigen()) <= bound_y && eq_inf <= -bound_y &&
512 infty_norm(CTdz.to_eigen()) <= bound_z && in_inf <= -bound_z;
513 return res;
514}
532template<typename T, typename I, typename P>
533bool
534global_dual_residual_infeasibility(VectorViewMut<T> Adx,
535 VectorViewMut<T> Cdx,
536 VectorViewMut<T> Hdx,
537 VectorViewMut<T> dx,
538 QpView<T, I> qp_scaled,
539 const Settings<T>& qpsettings,
540 const Model<T, I>& qpmodel,
541 P& ruiz)
542{
543
544 // The problem is dual infeasible if one of the conditions hold:
545 //
546 // FIRST
547 // ||unscaled(Adx)|| <= eps_d_inf ||unscaled(dx)||
548 // unscaled(Cdx)_i \in [-eps_d_inf,eps_d_inf] ||unscaled(dx)|| if u_i and l_i
549 // are finite or >=
550 // -eps_d_inf||unscaled(dx)|| if u_i = +inf or <= eps_d_inf||unscaled(dx)|| if
551 // l_i = -inf
552 //
553 // SECOND
554 //
555 // ||unscaled(Hdx)|| <= c eps_d_inf * ||unscaled(dx)|| and q^Tdx <= -c
556 // eps_d_inf ||unscaled(dx)||
557 // the variables in entry are changed in place
558 ruiz.unscale_dual_residual_in_place(Hdx);
559 ruiz.unscale_primal_residual_in_place_eq(Adx);
560 ruiz.unscale_primal_residual_in_place_in(Cdx);
561 T gdx = (dx.to_eigen()).dot(qp_scaled.g.to_eigen());
562 ruiz.unscale_primal_in_place(dx);
563
564 T bound = infty_norm(dx.to_eigen()) * qpsettings.eps_dual_inf;
565 T bound_neg = -bound;
566
567 bool first_cond = infty_norm(Adx.to_eigen()) <= bound;
568
569 for (i64 iter = 0; iter < qpmodel.n_in; ++iter) {
570 T Cdx_i = Cdx.to_eigen()[iter];
571 if (qp_scaled.u.to_eigen()[iter] <= 1.E20 &&
572 qp_scaled.l.to_eigen()[iter] >= -1.E20) {
573 first_cond = first_cond && Cdx_i <= bound && Cdx_i >= bound_neg;
574 } else if (qp_scaled.u.to_eigen()[iter] > 1.E20) {
575 first_cond = first_cond && Cdx_i >= bound_neg;
576 } else if (qp_scaled.l.to_eigen()[iter] < -1.E20) {
577 first_cond = first_cond && Cdx_i <= bound;
578 }
579 }
580
581 bound *= ruiz.c;
582 bound_neg *= ruiz.c;
583 bool second_cond_alt1 =
584 infty_norm(Hdx.to_eigen()) <= bound && gdx <= bound_neg;
585 bound_neg *= qpsettings.eps_dual_inf;
586
587 bool res = first_cond && second_cond_alt1 && infty_norm(dx.to_eigen()) != 0;
588 return res;
589}

619template<typename T, typename I, typename P>
620auto
621unscaled_primal_dual_residual(
622 VecMapMut<T> primal_residual_eq_scaled,
623 VecMapMut<T> primal_residual_in_scaled_lo,
624 VecMapMut<T> primal_residual_in_scaled_up,
625 VecMapMut<T> dual_residual_scaled,
626 T& primal_feasibility_eq_rhs_0,
627 T& primal_feasibility_in_rhs_0,
628 T& dual_feasibility_rhs_0,
629 T& dual_feasibility_rhs_1,
630 T& dual_feasibility_rhs_3,
631 P& precond,
632 Model<T, I> const& data,
633 QpView<T, I> qp_scaled,
634 VecMap<T> x_e,
635 VecMap<T> y_e,
636 VecMap<T> z_e,
637 proxsuite::linalg::veg::dynstack::DynStackMut stack)
638 -> proxsuite::linalg::veg::Tuple<T, T>
639{
640 isize n = x_e.rows();
641
642 LDLT_TEMP_VEC_UNINIT(T, tmp, n, stack);
643 dual_residual_scaled = qp_scaled.g.to_eigen();
644 {
645 tmp.setZero();
646 noalias_symhiv_add(tmp, qp_scaled.H.to_eigen(), x_e);
647 dual_residual_scaled += tmp;
648
649 precond.unscale_dual_residual_in_place({ proxqp::from_eigen, tmp });
650 dual_feasibility_rhs_0 = infty_norm(tmp);
651 }
652
653 {
654 auto ATy = tmp;
655 ATy.setZero();
656 primal_residual_eq_scaled.setZero();
657
658 detail::noalias_gevmmv_add(
659 primal_residual_eq_scaled, ATy, qp_scaled.AT.to_eigen(), x_e, y_e);
660
661 dual_residual_scaled += ATy;
662
663 precond.unscale_dual_residual_in_place({ proxqp::from_eigen, ATy });
664 dual_feasibility_rhs_1 = infty_norm(ATy);
665 }
666
667 {
668 auto CTz = tmp;
669 CTz.setZero();
670 primal_residual_in_scaled_up.setZero();
671
672 detail::noalias_gevmmv_add(
673 primal_residual_in_scaled_up, CTz, qp_scaled.CT.to_eigen(), x_e, z_e);
674
675 dual_residual_scaled += CTz;
676
677 precond.unscale_dual_residual_in_place({ proxqp::from_eigen, CTz });
678 dual_feasibility_rhs_3 = infty_norm(CTz);
679 }
680 precond.unscale_primal_residual_in_place_eq(
681 { proxqp::from_eigen, primal_residual_eq_scaled });
682 primal_feasibility_eq_rhs_0 = infty_norm(primal_residual_eq_scaled);
683
684 precond.unscale_primal_residual_in_place_in(
685 { proxqp::from_eigen, primal_residual_in_scaled_up });
686 primal_feasibility_in_rhs_0 = infty_norm(primal_residual_in_scaled_up);
687
688 auto b = data.b;
689 auto l = data.l;
690 auto u = data.u;
691 primal_residual_in_scaled_lo =
692 positive_part(primal_residual_in_scaled_up - u) +
693 negative_part(primal_residual_in_scaled_up - l);
694
695 primal_residual_eq_scaled -= b;
696 T primal_feasibility_eq_lhs = infty_norm(primal_residual_eq_scaled);
697 T primal_feasibility_in_lhs = infty_norm(primal_residual_in_scaled_lo);
698 T primal_feasibility_lhs =
699 std::max(primal_feasibility_eq_lhs, primal_feasibility_in_lhs);
700
701 // scaled Ax - b
702 precond.scale_primal_residual_in_place_eq(
703 { proxqp::from_eigen, primal_residual_eq_scaled });
704 // scaled Cx
705 precond.scale_primal_residual_in_place_in(
706 { proxqp::from_eigen, primal_residual_in_scaled_up });
707
708 precond.unscale_dual_residual_in_place(
709 { proxqp::from_eigen, dual_residual_scaled });
710 T dual_feasibility_lhs = infty_norm(dual_residual_scaled);
711 precond.scale_dual_residual_in_place(
712 { proxqp::from_eigen, dual_residual_scaled });
713
714 return proxsuite::linalg::veg::tuplify(primal_feasibility_lhs,
715 dual_feasibility_lhs);
716}
