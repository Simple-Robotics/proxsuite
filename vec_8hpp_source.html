197template<typename T>
198auto
199relocate(void* out, void const* in, usize nbytes) noexcept -> void*
200{
201 T* out_T = static_cast<T*>(out);
202 T* in_T = const_cast<T*>(static_cast<T const*>(in));
203 usize n = nbytes / sizeof(T);
204
205 for (usize i = 0; i < n; ++i) {
206 mem::construct_at(out_T + i, static_cast<T&&>(in_T[i]));
207 in_T[i].~T();
208 }
209
210 return out;
211}
212
213template<typename T>
214auto
215relocate_backward(void* out, void const* in, usize nbytes) noexcept -> void*
216{
217 T* out_T = static_cast<T*>(out);
218 T* in_T = const_cast<T*>(static_cast<T const*>(in));
219 usize n = nbytes / sizeof(T);
220
221 for (usize i = 0; i < n; ++i) {
222 mem::construct_at(out_T + (n - i - 1), static_cast<T&&>(in_T[n - i - 1]));
223 in_T[n - i - 1].~T();
224 }
225
226 return out;
227}

246template<typename T, typename A, typename C>
247auto
248alloc_and_copy(RefMut<A> alloc, RefMut<C> cloner, T const* data, usize len)
249 VEG_NOEXCEPT_IF(VEG_CONCEPT(alloc::nothrow_alloc<A>) &&
250 VEG_CONCEPT(alloc::nothrow_clone<C, T, A>)) -> mem::AllocBlock
251{
252
253 mem::AllocBlock block = mem::Alloc<A>::alloc(
254 RefMut<A>(alloc), mem::Layout{ sizeof(T) * usize(len), alignof(T) });
255
256 // if copying fails, this takes care of deallocating
257 Defer<AllocCleanup<A>> _{ {
258 alloc,
259 block.data,
260 mem::Layout{ block.byte_cap, alignof(T) },
261 } };
262
263 // copy construct elements
264 _collections::slice_clone(_.fn.alloc,
265 cloner,
266 static_cast<T*>(block.data),
267 static_cast<T*>(block.data) + len,
268 data);
269
270 _.fn.data = nullptr;
271 return block;

274template<typename T, typename A, typename C>
275auto
276realloc_and_append( //
277 RefMut<A> alloc,
278 RefMut<C> cloner,
279 mem::AllocBlock out,
280 usize out_len,
281 T const* in,
282 usize in_len) VEG_NOEXCEPT_IF(VEG_CONCEPT(alloc::nothrow_grow<A>) &&
283 VEG_CONCEPT(alloc::nothrow_clone<C, T, A>))
284 -> mem::AllocBlock
285{
286
287 if (in_len == 0) {
288 return out;
289 }
290
291 if (out.byte_cap >= (in_len + out_len) * sizeof(T)) {
292 mem::AllocBlock block = mem::Alloc<A>::grow(
293 RefMut<A>(alloc),
294 static_cast<void*>(out.data),
295 mem::Layout{ out.byte_cap, alignof(T) },
296 out_len * sizeof(T),
297 mem::RelocFn{ collections::relocate_pointer<T>::value });
298
299 // if copying fails, this takes care of deallocating
300 Defer<AllocCleanup<A>> _{ {
301 alloc,
302 block.data,
303 mem::Layout{ block.byte_cap, alignof(T) },
304 } };
305 // if copying fails, this takes care of destroying
306 Defer<Cleanup<T, A, C>> destroy{ {
307 _.fn.alloc,
308 cloner,
309 static_cast<T*>(block.data),
310 static_cast<T*>(block.data) + out_len,
311 } };
312
313 // copy construct elements
314 _collections::slice_clone( //
315 destroy.fn.alloc,
316 destroy.fn.cloner,
317 static_cast<T*>(block.data) + out_len,
318 static_cast<T*>(block.data) + in_len,
319 in);
320 // disable destruction
321 destroy.fn.ptr = nullptr;
322 destroy.fn.ptr_end = nullptr;
323 // disable deallocation
324 _.fn.data = nullptr;
325 out = block;
326 } else {
327 // copy construct elements
328 _collections::slice_clone( //
329 alloc,
330 cloner,
331 static_cast<T*>(out.data) + out_len,
332 static_cast<T*>(out.data) + in_len,
333 in);
334 }
335 return out;
336}

341template<>
342struct CloneFromImpl<false>
343{
344 template<typename T, typename A, typename C>
345 static void fn(RefMut<A> lhs_alloc,
346 RefMut<C> cloner,
347 vector::RawVector<T>& lhs_raw,
348 Ref<A> rhs_alloc,
349 vector::RawVector<T> const rhs_raw)
350 VEG_NOEXCEPT_IF(VEG_CONCEPT(nothrow_copy_assignable<A>) && //
351 VEG_CONCEPT(alloc::nothrow_alloc<A>) && //
352 VEG_CONCEPT(alloc::nothrow_clone<C, T, A>) && //
353 VEG_CONCEPT(alloc::nothrow_clone_from<C, T, A>))
354 {
355
356 vector::RawVector<T> lhs_copy = lhs_raw;
357 usize rhs_len = (rhs_raw.end - rhs_raw.data);
358
359 if (!(lhs_alloc == rhs_alloc)) {
360 T* data = lhs_copy.data;
361 T* data_end = lhs_copy.end;
362
363 // clean up old alloc
364 _collections::backward_destroy(lhs_alloc, cloner, data, data_end);
365
366 // assign before deallocation in case it fails
367 lhs_raw = {};
368 lhs_copy = {};
369
370 // don't need to deallocate on backward_destroy failure, since lhs can
371 // still access and reuse the allocation
372 mem::Alloc<A>::dealloc( //
373 RefMut<A>(lhs_alloc),
374 static_cast<void*>(data),
375 mem::Layout{ (lhs_copy.end_alloc - lhs_copy.data) * sizeof(T),
376 alignof(T) });
377 }
378
379 lhs_alloc.get() = rhs_alloc.get();
380
381 if (lhs_raw.data == nullptr) {
382 usize len = rhs_raw.end - rhs_raw.data;
383
384 mem::AllocBlock blk =
385 _collections::alloc_and_copy(lhs_alloc, cloner, rhs_raw.data, len);
386 T* data = static_cast<T*>(blk.data);
387 lhs_raw = {
388 data,
389 data + len,
390 data + blk.byte_cap / sizeof(T),
391 };
392 return;
393 }
394
395 usize assign_len = _detail::min2(lhs_copy.len(), rhs_raw.len());
396 // copy assign until the shared len
397 _collections::slice_clone_from( //
398 lhs_alloc,
399 cloner,
400 lhs_copy.data,
401 lhs_copy.data + assign_len,
402 rhs_raw.data);
403
404 // destroy from the shared len until end of lhs
405 lhs_raw.end = lhs_raw.data + assign_len;
406 _collections::backward_destroy( //
407 lhs_alloc,
408 cloner,
409 lhs_copy.data + assign_len,
410 lhs_copy.end);
411
412 // pass allocation to realloc_and_append
413
414 lhs_raw = {};
415 // realloc and copy construct new elements until end of rhs
416 mem::AllocBlock block = _collections::realloc_and_append(
417 lhs_alloc,
418 cloner,
419 mem::AllocBlock{
420 lhs_copy.data,
421 (lhs_copy.end_alloc - lhs_copy.data) * sizeof(T),
422 }, // out
423 assign_len, // out_len
424 rhs_raw.data + assign_len, // in
425 rhs_len - assign_len); // in_len
426
427 lhs_raw = vector::RawVector<T>{
428 static_cast<T*>(block.data),
429 static_cast<T*>(block.data) + rhs_len,
430 static_cast<T*>(block.data) + block.byte_cap / sizeof(T),
431 };
432 }
433};
434template<>
435struct CloneFromImpl<true>
436{
437 template<typename T, typename A, typename C>
438 static void fn(RefMut<A> lhs_alloc,
439 RefMut<C> cloner,
440 vector::RawVector<T>& lhs_raw,
441 Ref<A> rhs_alloc,
442 vector::RawVector<T> const rhs_raw)
443 VEG_NOEXCEPT_IF(VEG_CONCEPT(nothrow_copy_assignable<A>) && //
444 VEG_CONCEPT(alloc::nothrow_dealloc<A>) &&
445 VEG_CONCEPT(alloc::nothrow_alloc<A>))
446 {
447
448 vector::RawVector<T> lhs_copy = lhs_raw;
449

451 (lhs_copy.cap() < rhs_raw.len()));
